\c 1 
\s حضرت یُونسؑ دا تَرسِیس دو لڳا وںڄَݨ
\p ﷲ سئِیں دی طرفُوں حضرت یُونسؑ بِن متّیٰ کُوں حُکم مِلیا جو ”اُٹّھی! وݙّے شہر نَینویٰ دو وںڄ اتے اُتّھو دے لوکیں کُوں خبردار کر ݙے جو اُنہیں دے بُرے کم ٻہُوں زیادہ وَدھ ڳَے ہِن۔“ پر ﷲ سئِیں دے حُکم دے برخلاف حضرت یُونسؑ عِراق دے شہر نَینویٰ وںڄَݨ دی بجائے اُلٹے پاسے مُلکِ سپین دی بںدَرگاہ تَرسِیس شہر دو نِکِل وںڄَݨ دے خیال نال اپݨے مُلک دی بںدَرگاہ یافا تے وںڄ پہُںچے۔ اُتّھاں ہِک پاݨی والا تِجارتی جہاز تَرسِیس دو وںڄَݨ واسطے تیار کھڑا ہا۔ حضرت یُونسؑ کِرایا ݙے کے ملاحیں دے نال ہُوں جہاز تے روانہ تھی پئے تاں جو فرمانِ اِلٰہی کُوں پُورا نہ کرݨا پووے۔
\p ول تِھیا اِیں جو ﷲ سئِیں دی طرفُوں سمںدر وِچ ٻہُوں وݙّا طُوفان آ ڳیا اِتنا جو جہاز دے غرق تِھیوَݨ دا خطرہ پئے ڳیا۔ تاں ملاح گھٻرا ڳَے اتے ہر ہِک اپݨے اپݨے مَعبُود دے اڳُّوں مدَد واسطے فریاد کرَݨ لڳ پیا۔ ول جہاز کُوں ہَلکا کرَݨ دی خاطِر ٻہُوں سارا سامان چا کے سمںدر وِچ سٹ ݙِتّا ڳیا۔
\p ہِیں دَوران حضرت یُونسؑ جہاز دے تلویں حِصّے وِچ لہہ کے بے‌فِکری دی نِںدر سُتّے پئے ہَن۔ جہاز دے کپتان حضرت یُونسؑ کُوں جَڳا کے آکھیا: ”تُوں کِہجی نِںدر وِچ غرق ہَئِیں؟ اُٹّھی! اپݨے مَعبُود دے اڳُّوں فریاد کر۔ شاید او ساݙے اُتّے رحم چا کرے اتے اسّاں غرق تِھیوَݨ کنُوں بچ وںڄُوں۔“ ول ملاحیں آپس وِچ صلاح کِیتی جو ”آؤ! اسّاں قُرعہ‌اںدازی کرُوں تاں جو پتّا لڳے جو ایہ آفت کَیںدی وجہ کنُوں ساݙے اُتّے آئی ہے۔“ اُنہیں قُرعہ‌اںدازی کِیتی تاں قُرعہ حضرت یُونسؑ دے ناں نِکلیا۔ ول اُنہیں حضرت یُونسؑ کُوں آکھیا: ”تُوں کَون تِھیںدِیں؟ کِتّھو اَئِیں؟ کیا کم کریںدِیں؟ کیا تَیں کُوئی اِہجا غلط کم کِیتے جَیںدی وجہ کنُوں ایہ مُصِیبَت آ ڳئی ہے؟“
\p حضرت یُونسؑ آکھیا: ”میݙی قَوم عِبرانی ہے۔ ﷲ سئِیں ، نِیلی چھتری والا، جیرھا بحر و بَر دا خالِق ہے، مَیں ہُوںدا منَّݨ والا ہاں۔“
\p حضرت یُونسؑ اُنہیں کُوں سارا حال‌حوال ݙے ݙِتّا۔ ہُݨ اُنہیں کُوں چںڳی طرح تے پتّا لڳ ڳیا جو اِیہو ہے جیرھا ﷲ سئِیں کُوں کںڈ کِیتی ودّے، تاں سارے دے سارے ٻہُوں زیادہ خَوف زَدہ تھی ڳَے اتے آکھݨ لڳ پئے: ”تَیں ایہ کیا کِیتے؟“ ول آکھیونے: ”اچّھا ہُݨ ایہ ݙسا جو کیا کِیتا وںڄے تاں جو ایہ ودھدا ہویا طُوفان رُک وںڄے؟“ حضرت یُونسؑ آکھیا: ”میکُوں یقِین ہے جو تُسّاں اِیں وݙّے طُوفان وِچ میݙی وجہ کنُوں پَھسِّیے پئے ہِوّے۔ لِہٰذا میکُوں چا کے سمںدر وِچ بُھکا مارو تاں طُوفان آپیں رُک ویسی۔“
\p ول وی ملاحیں وݙّا زور لایا جو کہِیں طرَحویں کِنارے تَئِیں پہُںچ وںڄِن پر نہ پہُںچ سَڳِّیے کِیُوں جو سمںدر دِیاں چھولیاں جُوں جُوں وَدھدِیاں ویںدِیاں ہَن۔ تَھک ہار کے اُنہیں بارگاہِ اِلٰہی وِچ فریاد کِیتی: ”ﷲ سائِیاں! اِیں شخص دی وجہ کنُوں ساکُوں غرق نہ کرِیں اتے اِیںدا خُون وی ساݙی گردَن تے نہ آوے۔ تُوں جو چاہںدِیں او کریںدِیں۔“ ول اُنہیں حضرت یُونسؑ کُوں چا کے سمںدر وِچ اُچِّھل سٹیا۔ طُوفان ہُوں ویلے رُکّیا ڳیا۔ ایہ ݙیکھ کے سارِئیں دے دِلیں وِچ ﷲ سئِیں دا خَوف ٻہہ ڳیا۔ ول اُنہیں بارگاہِ اِلٰہی وِچ ہِک قُربانی پیش کِیتی اتے مَنوتِیاں مَنیونے۔
\c 2
\s حضرت یُونسؑ دا دُعا مںگَݨ
\p ﷲ سئِیں ہِک ٻہُوں وݙّی مچّھی کُوں مُقرَّر کِیتا ہویا ہا جو حضرت یُونسؑ کُوں نِڳِل وںڄے اتے حضرت یُونسؑ ترائے ݙِیںہہ مچّھی دے پیٹ وِچ رہ ڳَے۔ حضرت یُونسؑ مچّھی دے پیٹ وِچ ہَن جݙّاں اُنہیںؑ ﷲ سئِیں کنُوں ایہ دُعا مںگی ہئی:
\q ”ﷲ سائِیاں! جیرھے ویلے مَیں مُصِیبَت وات آیا پیا ہم، مَیں تیݙے اڳُّوں فریاد کِیتی ہئی اتے تَیں میݙی فریاد سُݨ گِھدّی ہئی۔
\q مَیں تَحتُ الثّریٰ کنُوں مدَد واسطے تیݙے اڳُّوں فریاد کِیتی ہئی اتے تَیں عرشِ مُعَلّٰی تے میݙی فریاد سُݨ گِھدّی ہئی۔
\q تَیں میکُوں سمںدر دے تَل تَئِیں لہا چھوڑیا۔ میݙے چَو‌چدھار پاݨی اِی پاݨی ہا۔
\q سارِیاں چھولیاں اتے چَھلّاں میݙے سِر دے اُتُّوں گُزر ڳِیاں ہَن۔
\q اُوں ویلے میکُوں اِیویں لڳدا ہا جو تیݙی رَحمَت کنُوں محرُوم تھی ڳیا ہاں۔
\q مَیں تیݙے مُقَدَّس گھر کُوں ول کِیویں ݙیکھساں؟
\q مَیں پاݨی وِچ غرق تھی ڳیا ہم۔ میݙا ساہ گُھٹِیجَݨ لڳ پیا ہا۔
\q سمںدری ٻُوٹِیاں میݙے سِر تے وِلھٹِیج ڳِیاں ہَن۔
\q مَیں تاں سمںدری پہاڑیں دی تہہ وِچ او تَئِیں غرق تھی ڳیا ہم جِتُّھوں اڄ تَئِیں کُوئی واپس نَئِیں ولیا۔
\q ول وی ﷲ سائِیاں! تُوں میکُوں تَحتُ الثّریٰ کنُوں زِںدہ کڈھ اَئِیں۔
\q جیرھے ویلے میکُوں اپݨی زِںدگی دی بِالکُل اُمِید نہ رہ ڳئی ہئی تاں ﷲ سائِیاں! مَیں تیݙے اڳُّوں فریاد کِیتی ہئی،
\q اتے تَیں اپݨی بارگاہ وِچ میݙی فریاد سُݨ گِھدّی ہئی۔
\q جیرھے لوک ٻِنّہیں معبُودیں کُوں مَنیںدِن او تیݙی رَحمَت کنُوں محرُوم تھی ویںدِن۔
\q مَیں تیݙا شُکر ادا کریساں، تیݙی حمد کریساں اتے تیݙی بارگاہ وِچ قُربانِیاں پیش کریساں۔
\q اپݨِیاں مَنتاں مَنوتِیاں پُورِیاں کریساں۔ ﷲ سائِیاں! بچاوَݨ والا صِرف تُوں اِی ہَئِیں۔“
\p ﷲ سئِیں اُوں مچّھی کُوں حُکم ݙِتّا اتے مچّھی حضرت یُونسؑ کُوں سمںدر دے کِنارے تے اُڳِل سٹیا۔
\c 3
\s حضرت یُونسؑ دا ﷲ سئِیں دی فرماں برداری کرَݨ
\p ہِک دفعہ ولا ﷲ سئِیں دی طرفُوں حضرت یُونسؑ کُوں حُکم مِلیا جو ”اُٹّھی! اُوں وݙّے شہر نَینویٰ دو وںڄ اتے اُتّھو دے لوکیں کُوں اِیں ڳالھ کنُوں خبردار کر ݙے جیرھی مَیں تیکُوں ݙسیںدا پیاں۔“ ﷲ سئِیں دے حُکم دے مُطابِق حضرت یُونسؑ ہُوں ویلے اُٹّھی کے نَینویٰ دو لڳے ڳَے۔ نَینویٰ اِتنا وݙّا ہا جو پُورا شہر پِھرَّݨ واسطے ٹُردِئیں ٹُردِئیں ترائے ݙِیںہہ لڳ ویںدے ہَن۔ حضرت یُونسؑ شہر وِچ داخِل تھی کے پہلے ݙِیںہہ دا سفر شُروع کِیتا، مُختلِف جاہِیں تے اپݨے کولھ کَٹّھے تِھیوَݨ والے لوکیں کُوں زور شور نال ایہ تبلِیغ کریںدے رہ ڳَے جو ”چالِیہہ ݙِیںہہ دے بعد نَینویٰ شہر تباہ و برباد کر ݙِتّا ویسی۔“ نَینویٰ دے لوکیں ادنیٰ اعلیٰ، چھوٹِئیں وݙِّئیں سارِئیں اِیں پَیغامِ اِلٰہی دا یقِین کر گِھدا، ماتمی لِباس پا گِھدونے اتے خاک تے ٻہہ کے گِریہ زاری شُروع کر ݙِتونے۔ روزے رکَّھݨ دا اعلان کر ݙِتّا ڳیا۔
\p تِھیا اِیں جو ایہ خبر نَینویٰ دے بادشاہ تَئِیں وی وںڄ پہُںچی۔ تاں او ہُوں ویلے تخت کنُوں لتّھا آیا اتے شاہی لِباس لہا کے ماتمی لِباس پا گِھدُّس اتے خاک تے ٻہہ کے گِریہ زاری شُروع کر ݙِتُّس۔ ول بادشاہ اتے اُوںدے وزِیریں مُشِیریں دی طرفُوں سارے شہر وِچ ایہ اعلان کرا ݙِتّا ڳیا جو ”اڄ کنُوں روزے شُروع ہِن۔ نہ صِرف اِنسان، بلکہ اُنہیں دے پالتُو جانوَر وی کُوئی چِیز نہ کھاوِن پِیوِن، ہر کہِیں دے بدَن تے ماتمی لِباس ہووے اتے خاک تے ٻہہ کے گِریہ زاری کرو۔ ظُلم کنُوں باز آ وںڄو،بُرے طَور طرِیقے چھوڑ ݙیوو، کیا پتّا رَبُّ العِزَّت رحم چا فرماوے، اُوںدا قہَر نازِل نہ تِھیوے اتے اسّاں تباہ و برباد تِھیوَݨ کنُوں بچ وںڄُوں؟“
\c 4
\s رحمتِ اِلٰہی
\p جیرھے ویلے رَبُّ العِزَّت اُنہیں دی ایہ حالت ݙِٹّھی جو او اپݨے اپݨے بُرے چال چلَݨ کنُوں تَوبہ تائب تھی ڳَے ہِن تاں رَبُّ العِزَّت کُوں اُنہیں تے ترس آ ڳیا اتے اُنہیں تے عذاب نازِل نہ کِیتُس۔
\p پر حضرت یُونسؑ کُوں ایہ بِالکُل چںڳاں نہ لڳا اتے غُصّے کنُوں رَتّے لال تھی ڳَے۔ ہِیں واسطے اُنہیںؑ آکھیا: ”ﷲ سائِیاں! تَیں اُوہو کُجھ کِیتے جیرھا کُجھ مَیں اُوں ویلے سوچیا ہا جݙّاں مَیں اپݨے وطن تے ہم۔ ہِیں وجہ کنُوں تاں مَیں نَینویٰ وںڄَݨ دی بجائے تَرسِیس دو ویںدا پیا ہم کِیُوں جو مَیں ڄاݨدا ہم جو تُوں رحِیم و کرِیم ذات ہَئِیں۔ وݙّے تحَمُّل والا اتے بِالکُل مہربان ہَئِیں۔ عذاب نازِل کرَݨ کنُوں دَرگُزر کریںدِیں۔ ﷲ سائِیاں! میکُوں ہُݨ مار گَھت، اِیں جِیوَݨ کنُوں تاں مَر وںڄَݨ بہتر ہے۔“ ﷲ سئِیں فرمایا: ”اِیںدے وِچ ناراض تِھیوَݨ والی کیرھی ڳالھ ہے؟“
\p ول حضرت یُونسؑ ایہ ݙیکَھݨ واسطے جو آخِر نَینویٰ شہر دا بݨدا کیا ہے، شہر کنُوں ٻاہر مشرقی پاسے ونٚڄ ٻیٹھے۔ اُتّھاں اُنہیں اپݨے واسطے ہِک چھپری بݨا کے اُونٚدی چھاں تلّے ٻَہہ رِیہے۔ ول اِیں تِھیا جو اللّٰہ رَبُّ الٛعِزَّتٛ دے حُکم نال یکدَم ہِک وَلھ تھی آئی اتے حضرت یُونسؑ دے اُتُّوں کِھنٚڈ ڳئی تاں جو او تپِّش دی تکلِیف کنُوں بچِّئے رہ ونٚڄِن۔ اُنہیں کُوں اِیں وَلھ دی وݙّی خُوشی تھئی۔ پر ݙُوجھے ݙِینٚہہ اللّٰہ سئِیں دے حُکم نال ہِک کِیڑا آ کے سویرے سویرے اُوں وَلھ دِیاں پاڑاں کھا ڳیا اتے او سُک ڳئی۔ جیرھے ویلے سِجھ چَڑھیا، تاں رَب مشرقی پاسُوں تَتّی بھا ہَوا گھولی۔ اُوں ہَوا نال حضرت یُونسؑ دا بدَن سڑَّݨ پَے ڳیا۔ تیز دُھپ نال اُنہیں دا سِر ݙلدا ہا۔ ہِیں وجہ کنُوں اُنہیں کُوں تَڑپا لڳ ڳیا۔ ول مَوت دے واسطے دُعا منٚگدِئیں ہوئیں آکھیونے: ”اِیں جِیوَݨ کنُوں تاں مَر ونٚڄَݨ بہتر ہے۔“ ول رَبُّ الٛعِزَّتٛ حضرت یُونسؑ کُوں فرمایا: ”اِیں وَلھ دی خاطِر ناراض تِھیوَݨ والی کیرھی ڳالھ ہے؟“ اُنہیں آکھیا: ”ہا! ڳالھ ہے۔ بَس ہُݨ مَیں مَر ونٚڄَݨ چاہنٚداں۔“ اللّٰہ سئِیں فرمایا: ”تیکُوں اِیں وَلھ دی اِتنی فِکر ہے جیکُوں نہ رَہائی، نہ اُونٚدے وݙّا تِھیوَݨ واسطے مِحنَت کِیتی؟ او تاں آپ کنُوں آپیں ہِکی رات وِچ تھی آئی ہے اتے وَلدی رات سُکّی ڳئی ہے۔ میکُوں اِتنے وݙّے شہر دی کُوئی فِکر نہ ہوسی؟ نَینویٰ وِچ ہِک لَکھ وِیہہ ہزار کنُوں زِیادہ او اِنسان رہنٚدِن جِنّہیں کُوں صحِیح اتے غلط دا پتّا اِی کَینی؟ اِنہیں کنُوں عِلاوَہ اُتّھاں ٻہُوں سارے بےزبان جانوَر وی ہِن۔ کیا ایہ سارے میݙی رحمَت کنُوں محرُوم رہ ونٚڄِن؟“
