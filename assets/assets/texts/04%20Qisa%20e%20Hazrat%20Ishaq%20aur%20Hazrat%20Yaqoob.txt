\c 1 
\s حضرت یعقُوبؑ اتے عِیصَو دی پیدائش
\p حضرت ابراہیمؑ دے جانَشِین پُتر حضرت اِسحاقؑ اتے اُنہیں دی اَولاد دا تذکرہ۔ جݙّاں حضرت اِسحاقؑ بی بی رِفقہ نال شادی کِیتی اُوں ویلے حضرت اِسحاقؑ دی عُمر چالِیہہ سال ہئی۔ بی بی رِفقہ فَدّان ارام شہر دے باشِںدے بَتُوئیل دی دِھی اتے لابن دی بھیݨ ہئی۔ حضرت اِسحاقؑ اپݨی ذال واسطے ﷲ سئِیں کنُوں ٻہُوں دُعا مںگدے رہ ڳَے کِیُوں جو بی بی کُوں ٻال نہ تِھیںدے ہَن۔ ﷲ سئِیں اُنہیں دی دُعا قبُول کِیتی اتے بی بی رِفقہ کُوں حمل تھی ڳیا۔ بی بی دے پیٹ وِچ جاڑے ٻال ہِک ٻے کنُوں پہلے ڄاوَݨ واسطے ہِک ٻے کُوں جَھلَّݨ لڳ پئے، تاں بی بی اپݨے دِل وِچ آکھیا: ”میݙے نال اِیں کِیُوں پیا تِھیںدے؟“ ہِیں سِلسلے وِچ بی بی ﷲ سئِیں کنُوں دُعا مںگی۔
\p ﷲ سئِیں فرمایا:
\q ”تیݙے پیٹ وِچ ݙُو قَومیں دے وݙّے ہِن۔
\q اِنہیں دی آل اَولاد ہِک ٻے دی مُخالِف ہوسی۔
\q ہِک قبِیلہ ݙُوجھے کنُوں طاقتوَر تھی ویسی۔
\q وݙّے دا قبِیلہ چھوٹے دے قبِیلے دی خِدمت کریسی۔“
\p جݙّاں ٻال ڄاوَݨ دا ویلا پُورا تِھیا تاں پتّا لڳا جو بی بی رِفقہ دے پیٹ وِچ جاڑے پُتر ہَن۔ جیرھا پُتر پہلے ڄایا اُوںدا رَںگ رتّا لال ہا اتے اُوںدے سارے جِسم تے لُوں اِی لُوں ہئی۔ تاں اُنہیں اُوںدا ناں عِیصَو یعنی ”ڄَتیلا“ رکھیا۔ ول اُوںدے بعد ݙُوجھا ڄایا تاں او عِیصَو دی کُھرّی کُوں ہتّھ گَھتّی پیا۔ اُوںدا ناں یعقُوبؑ یعنی ”پِیچھا کرَݨ والا“ رکھیونے۔ جݙّاں ایہ ٻال پَیدا تَھے اُوں ویلے حضرت اِسحاقؑ سٹّھ سال دے ہَن۔
\c 2
\s عِیصَو دا اپݨے جانَشِین ہووَݨ دے حقّ دی قدر نہ کرَݨ
\p ایہ ٻال جیرھے ویلے وݙّے تھی ڳَے تاں عِیصَو ماہِر شِکاری اتے کُھلِّئیں جاہِیں کُوں پسںد کرَݨ والا بݨیا، پر حضرت یعقُوبؑ سادہ مزاج اتے عام گھریلُو زِںدگی دے شَوقِین رہ ڳَے۔ حضرت اِسحاقؑ کُوں عِیصَو زیادہ پیارا لڳدا ہا کِیُوں جو عِیصَو اُنہیں کُوں شِکار مار کے کَھویںدا ہا، پر بی بی رِفقہ کُوں حضرت یعقُوبؑ زیادہ پیارے لڳدے ہَن۔
\p ہِک ݙِیںہہ جݙّاں حضرت یعقُوبؑ مَوہری دی دال پَکیںدے پئے ہَن ہُوں ویلے عِیصَو جںگل کنُوں تھکّا ترُٹّا آیا۔ اُوکُوں ݙاڈھے زور دی بُکّھ لڳی ودّی ہئی۔ حضرت یعقُوبؑ کُوں آکھیُس: ”جلدی نال اِیں لال لال دال وِچُّوں تھوڑی جہِیں میکُوں ݙے کِیُوں جو مَیں بُکّھ کنُوں مَرداں۔“ ہِیں واسطے اُوںدا ناں ادوم یعنی ”لال“ پئے ڳیا۔ حضرت یعقُوبؑ آکھیا: ”پہلے تُوں اِیںدے بدلے اپݨے جانَشِین ہووَݨ دا حقّ میکُوں ݙے۔“ عِیصَو آکھیا: ”ݙیکھ! مَیں مرَݨ والا ہاں۔ ول جانَشِین ہووَݨ دا حقّ میݙے کیرھے کم اَوسی؟“ پر حضرت یعقُوبؑ آکھیا: ”پہلے تُوں ہُݨیں میݙے سامݨے قَسم چا۔“ تاں عِیصَو قَسم چا کے اپݨا جانَشِین ہووَݨ دا حقّ حضرت یعقُوبؑ کُوں ݙے ݙِتّا۔ ول حضرت یعقُوبؑ اُوںدے بدلے اُوکُوں روٹی اتے دال ݙِتّی۔ اُوں کھادَھا پِیتا اتے اُٹّھی کے لڳا ڳیا۔ اِیں طرح عِیصَو اپݨے جانَشِین ہووَݨ دے حقّ دی قدر نہ کِیتی۔
\c 3
\s حضرت اِسحاقؑ دا جرار شہر دو وںڄَݨ
\p اُوں پہلے کال دے بعد جیرھا حضرت ابراہیمؑ دے وقت وِچ پیا ہا اِیں مُلک وِچ ہِک واری ول کال پئے ڳیا۔ تاں حضرت اِسحاقؑ فلستِیہ دے وݙّے شہر جرار دو لڳے ڳَے جِتّھاں بادشاہ رہںدا ہا۔ اُوں بادشاہ کُوں اَبُومَلِک سݙیا ویںدا ہا۔ حضرت اِسحاقؑ کُوں ﷲ سئِیں ݙِکھالی ݙے کے فرمایا : ”تُوں مِصر دو نہ وںڄِیں، بلکہ جِیویں مَیں تیکُوں آکھاں اُوویں کر۔ تُوں ہِیں مُلک دے اںدَر رہ وںڄ۔ مَیں تیݙے نال رہ ویساں اتے تیکُوں بَرکتاں ݙیںدا رہ ویساں۔ ایہ سارا مُلک تیکُوں اتے تیݙی آل اَولاد کُوں ݙے ݙیساں۔ مَیں جیرھا عہد تیݙے والِد ابراہیمؑ نال کِیتا ہا اُوکُوں ضرُور پُورا کریساں۔ مَیں تیکُوں تارِئیں جِتنی آل اَولاد ݙیساں۔ ایہ سارا مُلک مَیں تیݙی نسل کُوں ݙے ݙیساں۔ رُوئے زمِین دِیاں سارِیاں قَوماں تیݙی آل اَولاد دے وسِیلے برکَت پَیسِن۔ ایہ سب کُجھ مَیں اِیں واسطے کریساں کِیُوں جو ابراہیمؑ تیݙے والِد ہمیشہ میݙی فرماں برداری کِیتی ہئی، میݙے سارے حُکمیں تے عمل کِیتا ہَس۔“
\p ہِیں واسطے حضرت اِسحاقؑ ہُوں مُلک کنُوں ٻاہر نہ ڳَے، بلکہ اُتھاؤُں دے وݙّے شہر جرار دے کولھ وںڄ رِیہے۔
\p جݙّاں اُتّھو دے لوکیں حضرت اِسحاقؑ کنُوں بی بی رِفقہ دے بارے وِچ پُچھیا تاں اُنہیں آکھیا: ”او میݙی بھیݨ ہے۔“ بی بی رِفقہ کُوں اپݨی ذال آکھݨ کنُوں اِیں واسطے ݙر ڳَے جو ”کِتھائِیں ایہ لوک رِفقہ کُوں حاصِل کرَݨ واسطے میکُوں مار نہ گَھتِّن“، کِیُوں جو بی بی ٻہُوں سوہݨی ہئی۔
\p جݙّاں حضرت اِسحاقؑ کُوں اُتھائِیں رہںدِئیں ہوئیں کافی ݙِیںہہ تھی ڳَے تاں ول اِیں تِھیا جو اُتّھو دے بادشاہ اَبُومَلِک ہِک ݙِیںہہ اپݨے محل دی کھِڑکی وِچُّوں ٻاہر جو ݙِٹّھا تاں حضرت اِسحاقؑ بی بی رِفقہ نال پیار پئے کریںدے ہَن۔ ول حضرت اِسحاقؑ کُوں بادشاہ سݙوا کے آکھیا: ”جَیںدے بارے وِچ تَیں آکھیا ہا جو ’او تیݙی بھیݨ ہے‘، او تاں تیݙی ذال ہے۔ کیرھی وجہ کنُوں تَیں ایہ جُھوٹ ٻولیا ہا؟“ حضرت اِسحاقؑ جواب ݙِتّا: ”میکُوں ݙر جو کِتھائِیں میکُوں اُوںدی وجہ کنُوں مار نہ گھتیا وںڄے۔“ بادشاہ آکھیا: ”تَیں ساݙے نال ایہ کیا کِیتے؟ اِیں تاں تھی سڳدا جو اِنہیں لوکیں وِچُّوں کُوئی تیݙی ذال نال بُرا کم کر گِھندا۔ تاں اِیں بُرائی دا ذِمَّہ‌دار کَون ہووے ہا؟“ ول بادشاہ سارے لوکیں واسطے ایہ حُکم جاری کِیتا: ”جے کہِیں شخص اِیں آدمی یا اِیںدی ذال کُوں ہتّھ لایا تاں اُوکُوں مار گھتیا ویسی۔“
\p حضرت اِسحاقؑ اُوں مُلک وِچ فصل رَہائی اتے پہلے اِی سال اُنہیں دی پیداوار وݙّے زور دی تھئی کِیُوں جو رَبُّ العِزَّت اُنہیں کُوں برکَت بخشی ہئی۔ اُنہیں کُوں بَخت لڳدا ڳیا اتے اُنہیں دی عِزَّت وَدھدی ڳئی۔ ول او ٻہُوں وݙّے آدمی بݨ ڳَے۔ اُنہیں دے کولھ بھیݙاں ٻَکرِیاں، ڳَوِّیں ݙاںد اتے ٻہُوں سارے خادِم چاکر تھی ڳَے۔ تاں فلستی اُنہیں دے نال حسد کرَݨ لڳ پئے۔ فلستِئیں او سارے کُھوہ جیرھے اُنہیں دے والِد حضرت ابراہیمؑ دی زِںدگی وِچ اُنہیں دے نَوکریں کَھٹِّئے ہَن، مِٹّی نال پُور کے بںد کر ݙِتّے ہَن۔ حضرت اِسحاقؑ او سارے کُھوہ ولا کَھٹوائے اتے اُنہیں دے اُوہے ناں ولا رکِّھئے جیرھے حضرت ابراہیمؑ رکِّھئے ہَن۔
\p ول اَبُومَلِک حضرت اِسحاقؑ کُوں آکھیا: ”تُوں ساݙے کولُھوں لڳا وںڄ کِیُوں جو تُوں اسّاں کنُوں زور تھی ڳَئِیں۔“ تاں حضرت اِسحاقؑ اُتّھو روانہ تھی کے جرار دی وادی وِچ لڳے ڳَے اتے اُتھائِیں رہَݨ لڳ پئے۔
\p اُتّھاں کُھوہ کَھٹیںدِئیں ہوئیں اُنہیں دے نَوکریں پاݨی دی ہِک رَو لدّھی۔ اُوں پاݨی دے راہ تے جرار وادی دے مقامی مال چَراوَݨ والِئیں حضرت اِسحاقؑ دے مال چَراوَݨ والِئیں نال جھیڑا کِیتا اتے آکھیا: ”ایہ ساݙا ہے۔“ ہِیں واسطے حضرت اِسحاقؑ اُوں کُھوہ دا ناں ”وِڳڑی والا“ رکھ ݙِتّا کِیُوں جو ہُوںدے راہ تے جھیڑا تھی پیا۔ ول حضرت اِسحاقؑ دے نَوکریں ݙُوجھا کُھوہ کھٹِیا۔ اُوںدے اُتّے وی جھیڑا تھی پیا اتے دُشمَنی وَدھ ڳئی۔ تاں حضرت اِسحاقؑ اُوںدا ناں ”وَیری والا“ رکھ ݙِتّا۔
\p ول او اُتّھو کنُوں اڳُوں تے لڳے ڳَے اتے ہِک ٻیا کُھوہ وںڄ کَھٹیونے۔ اُوں کُھوہ دے راہ تے جھیڑا نہ تِھیا۔ تاں حضرت اِسحاقؑ آکھیا: ”ہُݨ رَبُّ العِزَّت ساݙے رہَݨ واسطے کُھلّی جاہ ݙے ݙِتّی ہے اتے اسّاں ہِیں مُلک وِچ وَدھدے پَھلدے رہ ویسُوں۔“ ہِیں واسطے اُوں کُھوہ دا ناں “رحوبوت” یعنی ”وَسَّݨ آباد“ رکھیونے۔
\c 4
\s ﷲ سئِیں دا حضرت اِسحاقؑ دے نال وعدہ
\p ول بعد وِچ اُتّھو حضرت اِسحاقؑ بِئرِ‌شَبَع دو ول ڳَے۔ پہلی رات اِی اُتّھاں ﷲ سئِیں اُنہیں کُوں ݙِکھالی ݙے کے فرمایا: ”مَیں او خُدا ہاں جیکُوں ابراہیمؑ تیݙا والِد منیںدا۔ گھٻرا ناں کِیُوں جو مَیں تیݙے نال ہاں۔ مَیں اپݨے بںدے ابراہیمؑ دی خاطِر تیکُوں بَرکتاں ݙیساں اتے تیݙی نسل کُوں وَدھیساں پَھلیساں۔“ اِیںدے بعد حضرت اِسحاقؑ اُتّھاں بارگاہِ اِلٰہی وِچ قُربانِیاں پیش کرَݨ واسطے ہِک تَھلّا بݨایا اتے دُعا مَںگیونے۔ ول اُتھائِیں اپݨا خَیمہ لایونے اتے اپݨے نَوکریں کنُوں ہِک کُھوہ کَھٹواوَݨ شُروع کِیتونے۔
\c 5
\s حضرت اِسحاقؑ اتے اَبُومَلِک دوم دے درمیان مُعاہِدہ
\p اَبُومَلِک بادشاہ اپݨے شہر جرار کنُوں اپݨے مُشِیر اخوزت اتے کماںڈر فیکول دے کَٹّھے حضرت اِسحاقؑ دے کولھ لںگھ آیا۔ تاں حضرت اِسحاقؑ اُنہیں کُوں آکھیا: ”تُسّاں تاں میکُوں چںڳاں نہ سمَجھدے ہاوے اتے میکُوں اپݨے کولُھوں کڈھ وی چھوڑیا ہِوّے، ہُݨ میݙے کولھ کِیویں آ ڳَے ہِوّے؟“ اُنہیں آکھیا: ”اسّاں چںڳی طرح ݙیکھ گِھدّے جو ﷲ سئِیں تیݙے نال ہے کِیُوں جو تیݙے اُتّے اُوںدِیاں بَرکتاں ہِن۔ اِیں واسطے ساکُوں خیال آئے جو اسّاں آپس وِچ ایہ مُعاہِدہ کر گِھنُوں جو جِیویں اسّاں تیکُوں کُوئی نُقصان نہ پہُںچایا تُوں وی ساکُوں کُوئی نُقصان نہ پہُںچیسیں کِیُوں جو اسّاں تیݙے نال چںڳاں سلُوک کریںدِئیں ہوئیں تیکُوں خَیر خُوشی نال روانہ کِیتا ہا۔“ ول حضرت اِسحاقؑ اُنہیں دے واسطے دعوت پَکوائی اتے سارِئیں مِل کے کَٹّھے کھادَھا پِیتا۔ اڳلے ݙِیںہہ سویرےسویرے اُٹّھی کے اُنہیں قَسم چا کے ایہ مُعاہِدہ کِیتا جو ”ہِک ٻے کُوں کُوئی نُقصان نہ پہُںچیسُوں۔“ اِیںدے بعد حضرت اِسحاقؑ اُنہیں کُوں خَیر خُوشی نال روانہ کر چھوڑیا۔
\p ہُوں ݙِیںہہ حضرت اِسحاقؑ دے نَوکریں آن ݙسایا جو جیرھا کُھوہ اسّاں کَھٹیںدے پئے ہاسے اُوںدے وِچُّوں پاݨی نِکِل آئے۔ اِیں واسطے حضرت اِسحاقؑ اُوں کُھوہ دا ناں ”قَسم والا“ رکھیا۔ ہِیں وجہ کنُوں اُوں شہر دا ناں ولا بِئرِ‌شَبَع یعنی ”قَسم والا کُھوہ“ پَکّا تھی ڳیا۔
\c 6
\s عِیصَو دا غَیریں وِچ شادی کرَݨ
\p عِیصَو جݙّاں چالِیہہ سالیں دا تِھیا تاں اُوں بنُوحِتّ وِچُّوں بِیری دی دِھی یہُودِتہ اتے ول ایلون دی دِھی بشامَتھ نال شادی کِیتی۔ اِنہیں ݙُہائیں حضرت اِسحاقؑ اتے بی بی رِفقہ دا جِیوَݨ حرام کر ݙِتّا۔
\c 7
\s حضرت اِسحاقؑ دا حضرت یعقُوبؑ کُوں برکَت واسطے دُعا ݙیوَݨ
\p حضرت اِسحاقؑ ٻُڈھڑے تھی ڳَے ہَن۔ اُنہیں دِئیں اکِّھیں تے اِتنی دُھںد آ ڳئی ہئی جو او چںڳی طرح تے ݙیکھ نہ سڳدے ہَن۔ ہِک ݙِیںہہ اُنہیں اپݨے وݙّے پُتر عِیصَو کُوں سݙیا: ”میݙا پُتر!“ اُوں جواب ݙِتّا: ”جی سئِیں!“ حضرت اِسحاقؑ آکھیا: ”ݙیکھ! مَیں ہُݨ ٻُڈھڑا تھی ڳِیاں اتے پتّا نَئِیں کیرھے ویلے مَوت آ وںڄے۔ اِیویں کر۔ تِیر کماݨ چا کے جںگل دو لڳا وںڄ اتے شِکار مار کے گِھن آ۔ ول میݙی پسںد دا مزیدار کھاݨا پَکا کے میݙے کولھ چا آ تاں جو مَیں اُوکُوں کھا کے مرَݨ کنُوں پہلے بارگاہِ اِلٰہی وِچ تیݙے واسطے تہہ دِلُوں برکتیں دی دُعا مںگاں۔“
\p ایہ سارِیاں ڳالِھیں بی بی رِفقہ وی سُݨدی پئی ہئی۔ بَس جیرھے ویلے عِیصَو اپݨے والِد واسطے شِکار مارَݨ جںگل دو لڳا ڳیا تاں بی بی رِفقہ حضرت یعقُوبؑ کُوں آکھیا: ”ݙیکھ میݙا پُتر! مَیں تیݙے والِد کُوں تیݙے بِھرا دے نال ڳالِھیں کریںدا ہویا سُݨِئے۔ او اُوکُوں پیا اَہدا ’شِکار مار آ۔ ول میݙی پسںد دا مزیدار کھاݨا پَکا تاں جو مَیں اُوکُوں کھا کے مرَݨ کنُوں پہلے بارگاہِ اِلٰہی وِچ تیݙے واسطے برکتیں دی دُعا مںگاں۔‘
\p اِیں واسطے ہُݨ تُوں اُوویں کر، میݙا پُتر! جِیویں مَیں تیکُوں آکھاں۔ ہُݨیں ہُݨیں وںڄ کے اجَڑ وِچُّوں ٻکری دے ݙُو چںڳے چںڳے لیلے پکِّڑ کے میکُوں آن ݙے اتے مَیں اُنہیں دے گوشت دا تیݙے والِد دی پسںد دے مُطابِق کھاݨا پَکا ݙیںدی ہاں۔ ول او کھاݨا تُوں اپݨے والِد دے اڳُّوں وںڄ رکِّھیں۔ تاں او برکتیں واسطے تیکُوں دُعا ݙیسی۔“ حضرت یعقُوبؑ اپݨی ماء کُوں آکھیا: ”عِیصَو دے جِسم تے وال ہِن، میݙے جِسم تے تاں کَینی۔ جے ابّے سئِیں میکُوں ٹٹولیا چا تاں مَیں اُنہیں دی نِگاہ وِچ دھوکے‌باز بݨ ویساں۔ ول ایہ نہ تِھیوے جو دُعا دی جاہ تے بَددُعا گِھن آواں۔“ بی بی رِفقہ آکھیا: ”میݙا پُتر! شالا او بَددُعا تیݙی جاہ تے میکُوں لڳے۔ بَس تُوں میݙی ڳالھ منّ اتے لیلے پکِّڑ کے میکُوں آن ݙے۔“ حضرت یعقُوبؑ لڳے ڳَے اتے ٻکری دے ݙُو لیلے پکِّڑ کے ماء کُوں آن ݙِتونے۔ بی بی رِفقہ اُنہیں دے گوشت نال حضرت اِسحاقؑ دی پسںد دا کھاݨا پَکایا۔ ول اپݨے وݙّے پُتر عِیصَو دے چںڳے چںڳے کپڑے جیرھے گھر لاتھے ہَن کڈھ کے چھوٹے پُتر حضرت یعقُوبؑ کُوں پاوَݨ واسطے ݙِتّے۔ اِیںدے بعد بی بی لیلِئیں دا چَم حضرت یعقُوبؑ دے ٻاہِیں اتے گردَن دے پِچھلے پاسے جِتّھاں وال کَینا ہَن ولھیٹ ݙِتّا۔ ول اُوہو مزیدار کھاݨا اتے روٹی چا کے حضرت یعقُوبؑ دے ہتّھیں وِچ ݙے ݙِتونے۔ حضرت یعقُوبؑ اپݨے والِد دے کولھ آ کے اواز وَٹا کے آکھیا: ”ابّا سئِیں!“ حضرت اِسحاقؑ جواب ݙِتّا: ”جی پُتر! کیرھا ہَئِیں؟“ حضرت یعقُوبؑ آکھیا: ”مَیں تُہاݙا وݙّا پُتر عِیصَو ہاں۔ جِیویں تُسّاں آکِھئے مَیں اُوویں کِیتے۔ ہُݨ اُٹّھی ٻَہو اتے میݙے شِکار والے گوشت وِچُّوں کُجھ کھاؤ اتے میکُوں تہ دِلُوں دُعائِیں ݙیوو۔“ حضرت اِسحاقؑ پُچھیا: ”میݙا پُتر! اِتنی جلدی تَیں کِیویں شِکار مار گِھدّے؟“ اُنہیں جواب ݙِتّا: ”ﷲ سئِیں مہربان تھی پئے۔“ حضرت اِسحاقؑ آکھیا: ”پُتر! میݙے نزدِیک تھی آ!“ حضرت اِسحاقؑ ٹٹول کے ایہ ݙیکَھݨ چاہںدے ہَن جو کیا ایہ واقعی عِیصَو ہے؟ جیرھے ویلے حضرت یعقُوبؑ نزدِیک تھی ڳَے تاں حضرت اِسحاقؑ اُنہیں کُوں ٹٹولیا اتے سوچݨ لڳ پئے جو ”اواز تاں یعقُوبؑ والی لڳدی ہے، پر ٻاہاں عِیصَو والیاں لڳدِن۔“
\p بَس او حضرت یعقُوبؑ کُوں سُںڄاݨ نہ سَڳِّئے کِیُوں جو اُنہیں دے ٻاہِیں تے عِیصَو واںگُوں وال ہَن۔ حضرت اِسحاقؑ دُعا ݙیوَݨ واسطے تیار تَھے تاں ہِک دفعہ ول حضرت یعقُوبؑ کنُوں پُچھیونے: ”کیا تُوں واقعی عِیصَو اِی ہَئِیں؟“ اُنہیں جواب ݙِتّا: ”جِیا سئِیں۔“ حضرت اِسحاقؑ آکھیا: ”اچّھا۔ کھاݨا میݙے کولھ گِھن آ۔ کھا کے تہ دِلُوں برکتیں واسطے تیکُوں دُعا ݙیواں۔“ حضرت یعقُوبؑ کھاݨا اُنہیں دے کولھ آن رکھیا اتے کُجھ آبِ اںگُور وی آن ݙِتونے۔ حضرت اِسحاقؑ او کھاݨا کھادَھا اتے آبِ اںگُور پِیتا۔ اِیںدے بعد اُنہیں آکھیا: ”پُتر! نزدِیک آ کے ذرا میکُوں ڳلکڑی تے پا۔“ حضرت یعقُوبؑ نزدِیک وںڄ کے اُنہیں کُوں ڳلکڑی پاتی تاں حضرت اِسحاقؑ کُوں عِیصَو دے کپڑِئیں دی خوشبو آئی۔ ول اُنہیں ایہ دُعا ݙِتّی:
\q ”میݙے پُتر دی خوشبو اُوں فصل واںگُوں ہے
\q جیکُوں ﷲ سئِیں رَںگ لایا۔
\q رَبُّ العِزَّت اسمان کنُوں پݨیال،
\q زمِین دی زرخیزی یعنی ڈھیر سارا اناج
\q اتے ٻہُوں سارا آبِ اںگُور تیکُوں بخشِئے۔
\q شالا قَوماں تیݙِیاں خِدمت‌گُزار تِھیوِن
\q اتے قبِیلے تیݙے اڳُّوں پئے جُھکِّن۔
\q شالا تُوں اپݨے بِھرا دا سردار بݨیں
\q اتے اُوںدی آل اَولاد تیݙے اڳُّوں پئی جُھکے۔
\q جیرھا تیکُوں بَددُعا ݙیوے، شالا ہُوکُوں لڳے۔
\q جیرھا تیکُوں دُعا ݙیوے اُوکُوں رَںگ لَڳِّن۔“
\p حضرت اِسحاقؑ کنُوں برکَت دِیاں دُعائِیں گِھن کے حضرت یعقُوبؑ ٻاہر نِکِل ڳَے تاں عِیصَو شِکار مار کے آ ڳیا۔ او وی مزیدار کھاݨا پَکا کے حضرت اِسحاقؑ دے کولھ چا آیا اتے آکھیُس: ”ابّا سئِیں! اُٹّھو! میݙے شِکار والے گوشت وِچُّوں کُجھ کھاؤ اتے تہ دِلُوں برکتیں واسطے میکُوں دُعا ݙیوو۔“ حضرت اِسحاقؑ پُچھیا: ”تُوں کیرھا ہَئِیں؟“ اُوں آکھیا: ”مَیں عِیصَو تے ہاں، تُہاݙا وݙّا پُتر۔“ ول تاں حضرت اِسحاقؑ کُوں زور دی کَمبڑِیں لڳ ڳئی اتے ٻہُوں حَیران تھی کے پُچھیونے: ”تے ول او کَون ہا جیرھا تیݙے آوَݨ کنُوں پہلے میکُوں شِکار دا گوشت کھوا کے دُعائِیں گِھن وی ڳَے؟ ہُݨ تاں ہُوں کُوں اِی بَرکتاں مِلدِیاں رہ ویسِن۔“ جݙّاں عِیصَو اپݨے والِد دِیاں ایہ ڳالِھیں سُݨیاں تاں ݙُکھ نال بھرِیج ڳیا۔ ٻاکاں مار  مار کے آکھݨ لڳ پیا: ”ابّا سئِیں! میکُوں وی دُعائِیں ݙیوو! میکُوں وی!!“ حضرت اِسحاقؑ آکھیا: ”تیݙا حقّ دھوکے نال تیݙا بِھرا گِھن ڳَے۔“ تاں عِیصَو آکھیا: ”اُوںدا ناں یعقُوبؑ یعنی ’حقّ مارَݨ والا‘ ٹِھیک اِی رکھیا ڳَے۔ تاہُوں تاں اُوں ݙُوجھی دفعہ میݙا حقّ مارِیے۔ جانَشِین ہووَݨ دا حقّ تاں اُوں مَیں کنُوں مار اِی گِھدا ہا اتے ہُݨ برکتیں واسطے میݙے حِصّے دِیاں دُعائِیں وی اُوہو گِھن ڳَے۔“ ول اُوں والِد کنُوں پُچھیا: ”کیا تُسّاں میݙے واسطے وی برکتیں دِیاں دُعائِیں مںگ سڳدے ہِوّے؟“
\p حضرت اِسحاقؑ، عِیصَو کُوں آکھیا: ”ہُݨ تاں مَیں اُوکُوں اِی تیݙا آقا بݨَݨ اتے اُوںدے سارے رِشتےداریں کُوں اُوںدا غُلام بݨَݨ دِیاں دُعائِیں ݙے ݙِتیاں ہِن۔ اناج اتے اںگُوریں دی فراوانی واسطے دُعائِیں وی ہُوکُوں ݙے ݙِتیاں ہِن۔ ہُݨ میݙا پُتر! تیݙے واسطے مَیں کیا مںگاں؟“ تاں عِیصَو آکھیا: ”بَس، بَرکتاں اِیہے کُجھ ہَن؟ ابّا سئِیں! میکُوں وی دُعا ݙیوو! ابّا سئِیں! میکُوں وی!!“ اِیہو کُجھ آکھ کے او ٻاکاں بھر بھر رووَݨ لڳ پیا۔ تاں حضرت اِسحاقؑ ٻولِئے:
\q ”ݙیکھ! بےشک تیݙے کولھ زرخیز زمِین کَینا ہوسی
\q پر تُوں تلوار دے زور تے اپݨا گُزارہ کِیتی رہ ویسیں۔
\q تُوں اپݨے بِھرا دا غُلام ہوسیں،
\q پر جݙّاں تیݙی بے‌قراری وَدھ ویسی
\q تاں تُوں اُوںدی غُلامی دِیاں زَںجِیراں تروڑ گھتیسیں۔“
\p اِیںدے بعد عِیصَو دے دِل وِچ حضرت یعقُوبؑ واسطے بُغض پئے ڳیا کِیُوں جو اُوںدے خیال وِچ حضرت یعقُوبؑ اُوںدا حقّ مار گِھدا ہا۔ عِیصَو سُچیںدا رہ ڳیا جو ”ابّے دے مرَݨ دے ݙِیہاڑے نزدِیک ہِن۔ اِیںدے بعد مَیں یعقُوبؑ کُوں مار گھتیساں۔“ جݙّاں بی بی رِفقہ کُوں اپݨے وݙّے پُتر عِیصَو دی نِیَّت دا پتّا لڳا تاں اپݨے چھوٹے پُتر حضرت یعقُوبؑ کُوں سݙ کے آکھیُس: ”ݙیکھ! عِیصَو تیکُوں قتل کر کے بدلہ گِھنَّݨ دے چکَّر وِچ ہے۔ اِیں واسطے میݙا پُتر! تُوں میݙی ڳالھ منّ اتے خاران دو اپݨے مامے لابن دے کولھ درُک وںڄ۔ جے تَئِیں تیݙے بِھرا دی کاوَڑ ٹھڈّھی نَئِیں تِھیںدی او تَئِیں اُتھائِیں گُزار۔ جیرھے ویلے اُوںدی کاوَڑ دی بھا وِسِّم ویسی اتے جیرھا کُجھ تَیں اُوںدے نال کِیتے اُوکُوں بُھل ویسی تاں مَیں تیکُوں اُتّھو سݙوا گِھنساں۔ اِتّھاں کہِیں ݙِیںہہ ایہ نہ تِھیوے جو مَیں تُسّاں ݙُہائیں کُوں وںڄا ٻَہواں۔“
\c 8
\s حضرت یعقُوبؑ دا اپݨے مامے لابن دے پاسے وںڄَݨ
\p بی بی رِفقہ حضرت اِسحاقؑ کُوں آکھیا: ”مَیں حِتّی قَوم دِئیں عَورتیں یعنی عِیصَو دِئیں ݙُہائیں ذالِیں کنُوں کَک تھئی پئی ہاں۔ اِنہیں تاں میݙا جِیوَݨ حرام کر ݙِتّے۔ جے یعقُوبؑ وی اِتھائِیں حِتّی قَوم دِئیں چھوہرِیں وِچُّوں کہِیں نال شادی کر گِھدّی تاں میݙے جِیوَݨ دی کیا چَس رہ ویسی؟“
\p حضرت اِسحاقؑ، حضرت یعقُوبؑ کُوں سݙوایا۔ دُعا سلام دے بعد اُنہیں کُوں تاکِید کِیتونے: ”تُوں کنعانی چھوہرِیں وِچُّوں کہِیں نال شادی نہ کرِیں، بلکہ اِیویں کر فَورًا اپݨے نانے بَتُوئیل دے گھر فَدّان ارام لڳا وںڄ! اُتھائِیں اپݨے مامے لابن دے دِھیرِیں وِچُّوں کہِیں نال شادی کر گِھنی۔ قادرِ مُطلق تیکُوں بَرکتاں ݙیوے، وَدھاوے پَھلاوے اتے تَیں کنُوں ٻہُوں سارے قبِیلِئیں والی ہِک وݙّی قَوم بݨ وںڄے۔ جیرھیاں بَرکتاں رَبُّ العِزَّت تیݙے ݙاݙے ابراہیمؑ کُوں بخشِیاں شالا او تیکُوں اتے تیݙے بعد تیݙی آل اَولاد کُوں بخشِئے تاں جو اِیہو مُلک جیرھا رَبُّ العِزَّت تیݙے ݙاݙے ابراہیمؑ کُوں بخشِیا اتے جَیںدے وِچ تُوں ہُݨ تَئِیں پردیسی تھی گُزریںدا پیا ہَئِیں تیݙی ملکِیَّت تھی وںڄے۔“ بَس حضرت اِسحاقؑ حضرت یعقُوبؑ کُوں فَدّان ارام دو روانہ کِیتا جِتّھاں لابن رہںدا ہا جیرھا ارام دے رہَݨ والے بَتُوئیل دا پُتر ہا، حضرت یعقُوبؑ اتے عِیصَو دی ماء بی بی رِفقہ دا بِھرا ہا۔
\c 9
\s عِیصَو دا ہِک ٻی شادی کرَݨ
\p عِیصَو کُوں پتّا لڳا جو ابّے سئِیں یعقُوبؑ کُوں دُعائِیں ݙے کے فَدّان ارام والے پاسے شادی کرَݨ واسطے روانہ کِیتے اتے ایہ وی جو دُعائِیں ݙیںدے ویلے اُوکُوں تاکِید کِیتی ہِنے جو ”تُوں کنعانی چھوہرِیں وِچُّوں کہِیں نال نہ شادی کرِیں۔“ ایہ وی پتّا لڳیُس جو حضرت یعقُوبؑ ماء پِئُو دی فرماں برداری کریںدِئیں ہوئیں فَدّان ارام دو لڳے وی ڳَے ہِن۔ تاں عِیصَو کُوں اِحساس تِھیا جو کنعانی چھوہرِیں میݙے ابّے کُوں چںڳِیاں نَئِیں لڳدِیاں، ہِیں واسطے او اپݨے چاچے حضرت اِسماعِیلؑ دے کولھ ٹُر ڳیا اتے اُنہیں دی دِھی مَحلَت نال شادی کر کے گھر گِھن آیا۔ مَحلَت نبایوط دی بھیݨ ہئی۔
\c 10
\s بَیت ایل وِچ حضرت یعقُوبؑ دا خواب ݙیکَھݨ
\p حضرت یعقُوبؑ بِئرِ‌شَبَع کنُوں روانہ تھی کے خاران دے پاسے ویںدے پئے ہَن۔ رستے وِچ ہِک جاہ تے اُنہیں کُوں رات پئے ڳئی۔ اُنہیں اُتھاؤُں دے پتھریں وِچُّوں ہِک پتَّھر چا کے سِرہاݨے واںگُوں سِر تلّے رکھیا اتے اُتھائِیں سُمھ تَھے۔ ول خواب وِچ پئے ݙِہدِن جو زمِین کنُوں گِھن کے اسمان تَئِیں ہِک پَوڑی بݨی کھڑی ہے جَیںدے اُتّے فرِشتے لَہںدے چَڑھدے کھڑِن۔ ول اُتُّوں ﷲ سئِیں دی اواز آئی: ”مَیں رَبُّ‌العالمِین ہاں جیکُوں ابراہیمؑ تیݙا ݙاݙا وی منیںدا اتے اِسحاقؑ تیݙا والِد وی منیںدے۔ اِیہا زمِین جَیںدے اُتّے تُوں سُتّا پَئِیں مَیں تیکُوں اتے تیݙی آل اَولاد کُوں ݙیساں۔ تیݙی آل اَولاد گَرد دے ذرِّئیں واںگُوں بے‌شُمار تھی ویسی اتے چارے پاسے کِھںڈ پُںڈ ویسی۔ دُنیا دِیاں سارِیاں قَوماں تیݙے اتے تیݙی نسل دے وسِیلے برکَت پَیسِن۔ ݙیکھ! بھاویں تُوں جِتّھاں وی وںڄیں مَیں تیݙے نال ہاں اتے تیݙی رکھوالی کریساں۔ مَیں تیکُوں ولا ہِیں مُلک وِچ گِھن آؤساں۔ جیرھا وعدہ مَیں تیݙے نال کِیتے اُوکُوں ضرُور پُورا کریساں۔ مَیں ہمیشہ تیݙے نال نال رہ ویساں۔“
\p ول حضرت یعقُوبؑ دی اکّھ کُھل ڳئی۔ اڄّاں رات ہئی۔ اُنہیں دِل وِچ آکھیا: ”یقِینًا اِیں جاہ تے ﷲ سئِیں مَوجُود ہے، پر میکُوں تے اِیں ڳالھ دا پتّا اِی کَینا ہا۔“ اُنہیں ݙر کے آکھیا: ”ایہ کِہجی ݙراکلی جاہ ہے۔ اِتّھاں یقِینًا رَبُّ العِزَّت دا گھر اتے اسمان دو وںڄَݨ دا دروازہ ہے۔“
\p حضرت یعقُوبؑ سویرےسویرے اُٹِّھیے۔ اُوں جاہ کُوں یاد رکَّھݨ واسطے ہُوں پتَّھر کُوں اُتھائِیں کھڑا کر کھڑایونے جیکُوں سِرہاݨے واںگُوں سِر تلّے رکّھی پئے ہَن۔ ول اُوں پتَّھر دے سِرے تے زَیتُون دا تیل سٹ کے اُوکُوں رَبُّ العِزَّت دے ناں نال مَنسُوب کر کھڑایونے اتے اِیں جاہ دا ناں بَیت ایل یعنی ”رَبُّ العِزَّت دا گھر“ رکھیونے۔ پہلے اِیں مقام دا ناں لُوز ہا۔ ول اُنہیں اُتھائِیں رَبُّ العِزَّت دے نال قَول اِقرار کِیتا: ”بےشک تُوں میݙے نال ہَئِیں اتے اِیں سفر وِچ میݙی رکھوالی کریںدا رہ ویسیں اتے میکُوں روٹی کپڑا ݙیںدا رہ ویسیں۔ جݙّاں مَیں خَیر خُوشی نال صحِیح+سلامَت ابّے سئِیں دے کولھ اپݨے گھر ول اَوساں، تاں اللّٰہ+سائِیاں! مَیں اپݨے آپ کُوں تیݙا نیک ٻانھاں بݨَݨ واسطے وقف کر ݙیساںاتے جیرھا کُجھ تَیں میکُوں بخشِیا ہوسی اُنہیں سارِئیں چِیزیں وِچُّوں ݙَہواں حِصَّہ ضرُور مَیں اِتّھاں تیݙے گھر وِچ آن ݙیساں جیکُوں یاد رکَھݨ واسطے مَیں ایہ پتَّھر کھڑا کر کھڑائے۔“ 
\c 11
\s حضرت یعقُوبؑ دا اپݨے مامے لابن دے گھر پہُںچَݨ
\p حضرت یعقُوبؑ سفر کریںدے ہوئے آخرکار مُلکِ کنعان دے مشرقی پاسے وںڄ پہُںچے۔ اُتّھاں اُنہیں مَیدان وِچ ہِک کُھوئی ݙِٹّھی جَیںدے نزدِیک بھیݙیں ٻَکرِئیں دے ترائے اجَڑ ٻیٹھے ہَن۔ آجڑی ہِیں کُھوئی کنُوں اپݨے اَجڑیں کُوں پاݨی پِلیںدے ہَن۔ اُوں کُھوئی دے مُوںہہ تے پتَّھر دا ہِک وݙّا پُڑ لاتھا۔ جیرھے ویلے سارے آجڑی اُتّھاں کَٹّھے تھی ویںدے ہَن تاں مِل کے اُوکُوں اُتّھو ہَٹیںدے ہَن اتے اپݨے اپݨے اجَڑ کُوں پاݨی پِلیںدے ہَن۔ بعد وِچ پتَّھر دے اُوں پُڑ کُوں کُھوئی دے مُوںہہ تے ولا رکھ ݙیںدے ہَن۔
\p حضرت یعقُوبؑ آجڑِئیں کنُوں پُچھیا: ”بِھراوو! تُسّاں کِتّھاں رہںدے ہِوّے؟“ اُنہیں جواب ݙِتّا: ”اسّاں اِتھائِیں خاران وِچ رہںدے ہَیں۔“ ولا پُچھیونے: ”کیا تُسّاں نَحُور دے پوترے لابن کُوں ڄاݨدے ہِوّے؟“ اُنہیں جواب ݙِتّا: ”جی ہا۔“ حضرت یعقُوبؑ آکھیا: ”اُنہیں دو خَیر‌سَلّا تاں ہے؟“ اُنہیں جواب ݙِتّا: ”جی ہا۔ اتے او ݙیکھ! اُوںدی دِھی راحِیلہ اپݨے اجَڑ دے نال اںدی پئی ہے۔“ حضرت یعقُوبؑ اُنھائیں نال ڳالِھیں کریںدے ہوئیں آکھیا: ”اڄّاں تاں ڈھیر سارا ݙِیںہہ کھڑے۔ فِی الحال اَجڑیں دے کَٹّھے تِھیوَݨ دا ویلا تاں کَینی تِھیا۔ تُسّاں اپݨے اَجڑیں کُوں پاݨی پِلا کے چَراوَݨ واسطے ولا گِھن وںڄو۔“ اُنہیں آکھیا: ”جے تَئِیں سارے آجڑی کَٹّھے نہ تھی وںڄِن او تَئِیں اسّاں اِیویں کر نَئِیں سڳدے کِیُوں جو پتَّھر دا ایہ پُڑ اِتنی باری ہے جو اسّاں سارے مِل کے اِیکُوں کُھوئی دے مُوںہہ کنُوں ہَٹیںدے ہَیں، ول اَجڑیں کُوں پاݨی پِلیںدے ہَیں۔“
\p حضرت یعقُوبؑ اڄّاں اُتھائِیں آجڑِئیں نال اِیہے ڳالِھیں کریںدے کھڑے ہَن تاں بی بی راحِیلہ اپݨے اجَڑ نال اُتھائِیں آ پہُںچی۔ جیرھے ویلے حضرت یعقُوبؑ اپݨی مُلیر بی بی راحِیلہ کُوں اتے اپݨے مامے لابن دے اجَڑ کُوں ݙِٹّھا تاں کلّـِہئیں وںڄ کے پتَّھر دے پُڑ کُوں کُھوئی دے مُوںہہ کنُوں ہَٹا چھوڑِیونے اتے اجَڑ کُوں پاݨی پِلوا گِھدونے۔ ول بی بی راحِیلہ کُوں مِلئے تاں خُوشی کنُوں اُنہیں کُوں رووَݨ آ ڳیا۔ آکھیونے: ”تیݙا ابّا میݙا ماما تِھیںدے، مَیں تیݙی بُؤا رِفقہ دا پُتر یعقُوبؑ ہاں۔“ تاں بی بی درُک کے اپݨے ابّے کُوں وںڄ ݙسایا۔
\p جیرھے ویلے لابن کُوں اپݨے بھݨیجے حضرت یعقُوبؑ دا پتّا لڳا تاں مِلَݨ واسطے جلدی جلدی ٹُردا آیا۔ ڳلکڑی پا کے اُنہیں دا سِر چُمیُس اتے گھر گِھن ڳیا۔ ول حضرت یعقُوبؑ اُوکُوں حال‌حوال ݙِتّا۔ لابن آکھیا: ”تُوں تاں میݙے اپݨے جِگر دا ٹُکڑا ہَئِیں۔“

\s حضرت یعقُوبؑ دی بی بی لیاہ اتے بی بی راحِیلہ نال شادی تِھیوَݨ
\p حضرت یعقُوبؑ کُوں اُتھائِیں رہںدِئیں ہوئیں مَہِینہ تھی ڳیا ہا، ول لابن حضرت یعقُوبؑ کُوں آکھیا: ”بےشک تُوں میݙا بَھݨیجا ہَئِیں، پر اِیںدا مطلب ایہ تاں نَئِیں جو تُوں مُفت وِچ میݙی خِدمت کریںدا رہ وںڄیں؟ میکُوں ݙسا جو مَیں تیݙی خِدمت دے بدلے وِچ تیکُوں کیا ݙیواں؟“
\p لابن دِیاں ݙُو دِھیرِیں ہَن۔ وݙّی دا ناں لیاہ اتے چھوٹی دا ناں راحِیلہ۔ بی بی لیاہ دِیاں اکِّھیں سوہݨِیاں ہَن، پر بی بی راحِیلہ ہر طرح نال ݙاڈھی سوہݨی ہئی۔ حضرت یعقُوبؑ بی بی راحِیلہ تے فِدا ہَن۔ اِیں واسطے اُنہیں آکھیا: ”جے تُسّاں راحِیلہ دی شادی میݙے نال کرَݨ دی حامی بَھرو تاں مَیں ست سال تُہاݙی خِدمت کریںدا رہ ویساں۔“ لابن آکھیا: ”ٹِھیک ہے۔ میکُوں تَیں کنُوں کُوئی ٻیا چںڳاں ہے؟“ بی بی راحِیلہ کُوں حاصِل کرَݨ واسطے حضرت یعقُوبؑ ست سال خِدمت کِیتی، پر محبَّت وِچ اُنہیں کُوں ایہ عرصہ ست ݙِیہاڑِئیں واںگُوں لڳا۔
\p اِیںدے بعد حضرت یعقُوبؑ مامے کُوں آکھیا: ”ہُݨ میݙی شادی کرو کِیُوں جو ست سال پُورے تھی ڳَے ہِن۔“ تاں لابن اُتّھو دے سارے لوکیں کُوں کاںڈھے ݙِتّے اتے شادی کر ݙِتُّس۔ لابن اپݨی ٻانھی بی بی زِلفہ وی اپݨی دِھی دی خِدمت واسطے نال ݙِتّی۔ شادی والی رات بی بی راحِیلہ دی جاہ تے لابن اپݨی وݙّی دِھی بی بی لیاہ کُوں حضرت یعقُوبؑ دی کُںوار بݨا ݙِتّا اتے گھوٹ کُںوار کَٹّھے سُتّے۔ سویرے جیرھے ویلے حضرت یعقُوبؑ ݙِٹّھا جو رات راحِیلہ دی جاہ تے میݙی کُںوار لیاہ ہئی تاں لابن دے کولھ وںڄ کے آکھیونے: ”تُسّاں میݙے نال ایہ کیا کِیتے؟ کیا مَیں اِتنا عرصہ تُہاݙی خِدمت راحِیلہ دے واسطے نَئِیں کریںدا رہ ڳیا؟ ول تُسّاں میݙے نال اِیں کِیُوں کِیتے؟“ لابن جواب ݙِتّا: ”حَوصلہ رکھ! اصَل وِچ ڳالھ ایہ ہے ساݙے مُلک دا ایہ رِواج کَینی جو وݙّی کنُوں پہلے چھوٹی دی شادی کر ݙیوُوں۔ اِیںدا ستّو‌واڑہ پُورا تِھیوے ول مَیں راحِیلہ دی شادی وی تیݙے نال کر ݙیساں۔ پر اُوںدی خاطِر تیکُوں ست سال ٻے میݙی خِدمت کرݨی پوسی۔“
\p حضرت یعقُوبؑ ایہ ڳالھ منّ گِھدّی۔ جݙّاں ستّو‌واڑہ پُورا تھی ڳیا تاں لابن اپݨی چھوٹی دِھی بی بی راحِیلہ دی شادی وی اُنہیں نال کر ݙِتّی۔ اتے اپݨی ٻانھی بی بی بِلہہ وی اپݨی دِھی بی بی راحِیلہ دی خِدمت واسطے نال ݙِتّی۔ تاں حضرت یعقُوبؑ بی بی راحِیلہ دے کَٹّھے وی وںڄ سُتّے۔ بی بی راحِیلہ اُنہیں کُوں بی بی لیاہ کنُوں ڈھیر پیاری لڳدی ہئی۔ ول او ست سال ٻے اپݨے مامے دی خِدمت کریںدے رہ ڳَے۔
\c 13
\s حضرت یعقُوبؑ دے ٻالیں دا ڄاوَݨ
\p ﷲ سئِیں ݙِٹّھا جو بی بی لیاہ کُوں پَے دا پیار نَئِیں مِلیا تاں بی بی کُوں اَولاد دے قابِل بݨایُس، پر بی بی راحِیلہ اَولاد کنُوں محرُوم رہ ڳئی۔ بی بی لیاہ کُوں پیٹ تھی ڳیا۔ ول اُوکُوں پُتر ڄایا۔ بی بی اُوںدا ناں روبَن یعنی ”ݙیکھو! ہِک پُتر“ رکھیا۔ بی بی لیاہ سوچیا: ”کِیُوں جو ﷲ سئِیں میݙے ݙُکھ دو ݙِٹّھے، ہِیں وجہ کنُوں ہُݨ مَیں اپݨے پَے کُوں پیاری لڳساں۔“ بی بی لیاہ کُوں ولا پیٹ تھی ڳیا۔ ول پُتر ڄایُس۔ تاں بی بی سوچیا جو ”ﷲ سئِیں میݙی دھاں سُݨ گِھدّی ہے جو میݙے نال پیار نَئِیں کِیتا ڳیا۔ اِیں واسطے میکُوں ٻیا وی پُتر بخشِیا ہِس۔“ ہِیں وجہ کنُوں بی بی اُوں پُتر دا ناں شِمعُون یعنی ”سُݨ گِھدّی“ رکھیا۔ ولا ٻیا پیٹ تھی ڳیا۔ ول وی پُتر ڄایا۔ تاں بی بی لیاہ سوچیا جو ”ہُݨ اِیں دفعہ میݙے پَے دا دِل میݙے نال لڳ ویسی کِیُوں جو مَیں کنُوں اُوںدے ترائے پُتر ڄَین۔“ اِیں واسطے بی بی اُوں پُتر دا ناں لیوی یعنی ”لڳ وںڄَݨ“ رکھیا۔ بی بی لیاہ کُوں ولا پیٹ تھی ڳیا اتے چَوتھا پُتر ڄایا۔ تاں بی بی اپݨے دِل وِچ آکھیا: ”ﷲ سئِیں بےشک وݙِّئیں شانیں والا ہے۔“ اِیں واسطے اُوں پُتر دا ناں یہُودَہ یعنی ”حمد“ رکھیُس۔ اِیںدے بعد ولا کُجھ عرصے تَئِیں بی بی لیاہ کُوں ٻال نہ تَھے۔
\p جݙّاں بی بی راحِیلہ ایہ ݙِٹّھا جو میکُوں اَولاد نَئِیں پئی تِھیںدی تاں بی بی کُوں اپݨی پہاج نال جیرھی اُوںدی وݙّی بھیݨ وی ہئی حسد تِھیوَݨ لڳ پیا۔ ول بی بی راحِیلہ حضرت یعقُوبؑ کُوں آکھیا: ”میکُوں وی اَولاد ݙے، نہ تاں مَیں مَر پوساں۔“ ایہ ڳالھ سُݨ کے حضرت یعقُوبؑ کُوں ݙاڈھے زور دا غُصّہ آ ڳیا۔ اُنہیں آکھیا: ”مَیں کُوئی خُدا ہاں جو تیکُوں اَولاد نِمّـِھیں پیا ݙیںدا۔“ بی بی راحِیلہ آکھیا: ”مَیں اپݨی ٻانھی بی بی بِلہہ تیکُوں بَخشیںدی ہاں۔ اُوںدی جیرھی اَولاد تِھیسی اُوکُوں مَیں اپݨی جھولی وِچ پا گِھنساں اتے اِیں طرح اَولاد والی تھی پوساں۔“ ہِیں واسطے بی بی راحِیلہ اپݨی ٻانھی بی بی بِلہہ دی شادی حضرت یعقُوبؑ نال کر ݙِتّی۔ بی بی بِلہہ کُوں پیٹ تھی ڳیا۔ ول اُوکُوں پُتر ڄایا۔ تاں بی بی راحِیلہ سوچیا جو ”رَبُّ العِزَّت میݙے حقّ وِچ فَیصلہ ݙے ݙِتّے، میݙی فریاد سُݨ کے میکُوں ہِک پُتر بخشِیا ہِس۔“ اِیں وجہ کنُوں بی بی اُوںدا ناں دان یعنی ”حقّ وِچ فَیصلہ“ رکھیا۔ بی بی راحِیلہ دی ٻانھی بی بی بِلہہ کُوں ولا پیٹ تھی ڳیا اتے ولا پُتر ڄایُس۔ تاں بی بی راحِیلہ اپݨے دِل وِچ آکھیا: ”مَیں اپݨی پہاج دے نال زور دی بھیڑی بِھڑّی ہاں اتے جِیت ڳئی ہاں۔“ ہِیں وجہ کنُوں بی بی اُوںدا ناں نفتالی یعنی ”بِھڑّݨ“ رکھیا۔
\p جݙّاں بی بی لیاہ ایہ محسُوس کِیتا جو میکُوں ٻال ڄاوَݨ وِچ دیر تھی ڳئی ہے تاں اِیں بی بی وی اپݨی ٻانھی بی بی زِلفہ دی شادی اپݨے پَے نال کر ݙِتّی۔ بی بی زِلفہ کُوں پیٹ تھی ڳیا۔ ول اُوکُوں پُتر ڄایا۔ بی بی لیاہ اُوںدا ناں جاد رکھیا یعنی ”میݙی قِسمَت جاڳی۔“ بی بی لیاہ دی ٻانھی بی بی زِلفہ کُوں ولا ٻیا پُتر ڄایا۔ تاں بی بی لیاہ سوچیا جو ”عَورتاں ہُݨ میکُوں بختاوَر آکھسِن۔“ اِیں واسطے اُوںدا ناں آشر یعنی ”بختاوَر“ رکھیونے۔
\p ول اِیں تِھیا جو کݨَک کپَّݨ دے ݙِیہاڑِئیں روبَن فصل وِچ ڳیا ودّا۔ اُوکُوں اُتھائِیں خاص قِسم دِیاں جڑی‌ٻُوٹِیاں لدِھیاں جِنہیں دے کھاوَݨ نال حمل تِھیوَݨ دی قُوَّت وَدھدی ہئی۔ او ٻُوٹِیاں اُوں اپݨی ماء بی بی لیاہ کُوں آن ݙِتیاں۔ تاں بی بی راحِیلہ بی بی لیاہ کُوں آکھیا: ”مہربانی کر کے اپݨے پُتر دِئیں جڑی‌ٻُوٹِئیں وِچُّوں تھوڑیاں جہاں میکُوں ݙے۔“ پر بی بی لیاہ جواب ݙِتّا: ”کیا ایہ تھوڑا ہے جو تَیں میݙا پَے مَیں کنُوں کَھس گِھدّے؟ ہُݨ میݙے پُتر دِیاں جڑی‌ٻُوٹِیاں وی مَیں کنُوں گِھنَّݨ چاہںدِیں؟“ تاں بی بی راحِیلہ آکھیا: ”جے تُوں جڑی‌ٻُوٹِیاں میکُوں ݙیویں چا تاں اُنہیں دے بدلے یعقُوبؑ اڄ دی رات تیݙے کولھ رہ ویسی۔“ شام کُوں جیرھے ویلے حضرت یعقُوبؑ فصل کنُوں وَلِّئے تاں بی بی لیاہ اڳُوں تے وَدھ کے اُنہیں کُوں آکھیا: ”اڄ دی رات تُوں میݙے کولھ رَہسیں کِیُوں جو مَیں اپݨے پُتر دِیاں جڑی‌ٻُوٹِیاں ݙے کے اڄ دی رات واسطے تیکُوں گِھن گِھدّے۔“ اِیں طرح او رات حضرت یعقُوبؑ بی بی لیاہ دے نال گُزاری۔ رَبُّ العِزَّت بی بی لیاہ دی دُعا سُݨی اتے بی بی کُوں پیٹ تھی ڳیا۔ اِیں طرح بی بی لیاہ کُوں پںجواں پُتر ڄایا۔ تاں بی بی لیاہ سوچیا: ”رَبُّ العِزَّت میکُوں ایہ اِنعام ݙِتّے کِیُوں جو مَیں اپݨی ٻانھی اپݨے پَے کُوں بخش ݙِتّی ہے۔“ ہِیں واسطے بی بی اُوں پُتر دا ناں اِشکار یعنی ”اِنعام“ رکھیا۔ جݙّاں بی بی لیاہ کُوں ولا پیٹ تِھیا تاں چھیواں پُتر ڄایا۔ ول بی بی سوچیا جو ”رَبُّ العِزَّت میکُوں قِیمتی نِعمَت نال نوازیا ہے۔ ہُݨ میݙا پَے میݙے نال رہ ویسی کِیُوں جو مَیں کنُوں اُوںدے چھی پُتر پَیدا تَھے ہِن۔“ اِیں واسطے بی بی لیاہ اُوں پُتر دا ناں زبُلُون یعنی ”نِعمَت“ رکھیا۔ کُجھ عرصے دے بعد بی بی لیاہ کنُوں ہِک دِھی پَیدا تھئی جَیںدا ناں دِینہ یعنی ”اِنصاف“ رکھیا ڳیا۔
\p آخرکار رَبُّ العِزَّت بی بی راحِیلہ دِیاں دُعائِیں سُݨ کے بی بی تے کرم دی نِگاہ فرمائی اتے ٻال ڄَݨَݨ دے قابِل بݨایُس۔ ول بی بی راحِیلہ کُوں پیٹ تھی ڳیا اتے پُتر ڄایا۔ تاں بی بی اپݨے دِل وِچ آکھیا: ”رَبُّ العِزَّت مَیں کنُوں میݨھاں لہا چھوڑے۔“ بی بی راحِیلہ اُوںدا ناں ’یُوسفؑ‘ رکھیا جَیںدا مطلب ہے: ”ﷲ سئِیں میکُوں ٻیا پُتر وی ݙیوے۔“

\s حضرت یعقُوبؑ دا امِیر تھی وںڄَݨ
\p بی بی راحِیلہ کنُوں حضرت یُوسفؑ دے ڄاوَݨ دے بعد حضرت یعقُوبؑ اپݨے سَوہرے لابن کُوں آکھیا: ”ہُݨ میکُوں اِجازَت ݙیوو تاں جو مَیں اپݨے وطن تے اپݨے گھر ول وںڄاں۔ میکُوں میݙے ٻال ٻَچّے نال گِھن وںڄَݨ دی وی اِجازَت ݙیوو جِنہیں دی خاطِر مَیں اِتنا عرصہ تُہاݙی خِدمت کریںدا رہ ڳِیاں۔ جِیویں مَیں دِل نال تُہاݙی خِدمت کِیتی ہے، اُوویں تُسّاں وی ہُݨ دِل نال میکُوں وںڄَݨ دی اِجازَت ݙیوو۔“ تاں لابن آکھیا: ”جے تیکُوں میݙا اِحساس ہے تاں اِتھائِیں رہ وںڄ کِیُوں جو مَیں فال کڈھی ہے اتے میکُوں پتّا لڳے جو ﷲ سئِیں تیݙی وجہ کنُوں میکُوں برکَت بخشی ہے۔“ ولا ایہ وی آکھیُس جو ”اڄ کنُوں اڳُوں تے مَیں تیکُوں مُعاوضہ ݙیںدا رہ ویساں جِتنا تُوں آکھیں۔“ حضرت یعقُوبؑ آکھیا: ”تُسّاں چںڳی طرح ڄاݨدے ہِوّے جو مَیں تُہاݙی کِیویں خِدمت کریںدا رہ ڳِیاں اتے کِیویں تُہاݙے مال کُوں سںبھالیا ہِم۔ میݙے آوَݨ کنُوں پہلے تُہاݙے کولھ تھوڑا جہاں مال ہا، پر ہُݨ وَدھ کے ڈھیر سارا تھی ڳَے۔ جݙّاں دا مَیں آیاں ﷲ سئِیں تُہاکُوں ٻہُوں برکَت بخشی ہے۔ ہُݨ میکُوں ݙساؤ مَیں اپݨے گھر دا بںدوبست کݙاں کراں؟“ تاں لابن آکھیا: ”اپݨا مُعاوضہ ݙَس!“ حضرت یعقُوبؑ آکھیا: ”تُسّاں میکُوں ٻیا کُجھ نہ ݙیوو۔ جے تُسّاں میݙی عرض قبُول کریںدے ہِوّے تاں مَیں ولا پہلے واںگُوں تُہاݙا اجَڑ چَریںدا رہ ویساں اتے اُنہیں دی سںبھال رکھیساں۔ عرض ایہ ہے جو اڄ مَیں وںڄ کے تُہاݙے اجَڑ کُوں ݙیکھساں بَھلیساں اتے اُنہیں وِچُّوں سارِیاں بَربَرِیاں، گَدریاں ٻَکرِیاں ٻَکرے اتے بُھوریاں بھیݙاں اتے گَھٹّے علیحدہ کر گِھنساں۔ بَس اِیہو کُجھ میݙا مُعاوضہ ہوسی۔ اڳُوں تے تُسّاں جݙّاں میݙے نال مُعاوضے دا حِساب کریسو تے میݙی اِیمانداری دا پتّا اِیں لڳسی جے میݙے کولھ کُوئی بھیݙ یا ٻکری اِنہیں رَںگیں کنُوں عِلاوہ ہُوئی تے مَیں تُہاݙا چور ہوساں۔“ تاں لابن آکھیا: ”جِیویں تَیں آکِھئے میکُوں منظُور ہے۔“
\p پر لابن خُود آپ ہُوں ݙِیںہہ وںڄ کے بَربَرے اتے گَدرے ٻَکرے ٻَکرِیاں یعنی او جِنہیں کُوں چِٹّے یا کالے چٹُک یا گَدر ہَن اتے سارِیاں بُھوریاں بھیݙاں اَںجھ کر کے اُنہیں کُوں اپݨے پُتریں دی نِگرانی وِچ ݙے ݙِتّا۔ ول اُوں اپݨے اتے حضرت یعقُوبؑ دے اَجڑیں دے درمیان ترائے ݙِیںہہ دا فاصلہ مُقرَّر کِیتا۔ اِیںدے بعد حضرت یعقُوبؑ اپݨے سَوہرے دے باقی اجَڑ کُوں چَریںدے رہ ڳَے۔ ول حضرت یعقُوبؑ سُفیدہ، بِدام اتے چنار دِیاں ساوِیاں لَڑیاں کپ کے اُنہیں دے اُتّے گَدر چِھلئے تاں لَڑِئیں دی سُفیدی ظاہِر تھی ڳئی۔ ول اُنہیں او لَڑیاں کھاݙِئیں اتے اَݙِّیں دے وِچ رکھ ݙِتیاں جِتّھاں اجَڑ پاݨی پِیوَݨ اںدے ہَن تاں جو پاݨی پِیںدے ویلے جو او ڳَبَّھݨ تِھیوِن تاں ایہ لَڑیاں اُنہیں دے اکِّھیں دے سامݨے ہووِن۔ اِنہیں لَڑِئیں دے سامݨے ڳَبَّھݨ تِھیوَݨ دی وجہ کنُوں اُنہیں گَدرے اتے بَربَرے پَھل ڄَݨئے۔ اِہجے پَھل حضرت یعقُوبؑ لابن دے اجَڑ وِچُّوں اپݨے واسطے نکھیڑ گِھندے ہَن۔ موٹے تازے مال دے لڳَّݨ دے ویلے حضرت یعقُوبؑ او لَڑیاں کھاݙِئیں وِچ رکھ ݙیںدے ہَن تاں جو مَستی دے ویلے او لَڑیاں اُنہیں دے اکِّھیں دے سامݨے ہووِن۔ پر سُکّے سڑے مال دے لڳَّݨ دے ویلے اُنہیں لَڑِئیں کُوں کھاݙِئیں وِچُّوں چا گِھندے ہَن۔ اِیں طرح سُکّا سڑیا مال لابن دا اتے موٹا تازہ مال حضرت یعقُوبؑ دا تِھیںدا ڳیا۔ ہِیں طرِیقے نال حضرت یعقُوبؑ دے اجَڑ وَدھدے ڳَے اتے او ٻہُوں سارے اُٹّھیں، گَݙہِیں، ٻانِھیاں اتے ٻانِھئیں دے مالِک بݨ ڳَے۔
\c 15
\s حضرت یعقُوبؑ دا مامے کنُوں چوری نِکلَݨ
\p حضرت یعقُوبؑ سُݨیا لابن دے پُتر ودّے اَہدِن جو ”یعقُوبؑ ساݙے والِد دا مال ولھیٹ کے ایہ ساری دَولَت کٹّھی کِیتی ہے۔“ اتے ایہ وی ݙِٹُّھس جو مامے لابن دا رَوِیّہ ہُݨ میݙے نال پہلے واںگُوں نَئِیں رہ ڳیا۔ تݙاہُوں ﷲ سئِیں حضرت یعقُوبؑ کُوں فرمایا: ”تُوں اپݨے پِئُو ݙاݙے دے مُلک دو اپݨے رِشتےداریں دے کولھ ول وںڄ اتے مَیں تیݙے نال ہوساں۔“ تاں حضرت یعقُوبؑ بی بی راحِیلہ اتے بی بی لیاہ کُوں اُتھائِیں سݙوا گِھدا جِتّھاں او مَیدان وِچ اجَڑ ٹِکائی پئے ہَن۔ اُنہیں کُوں آکھیونے: ”مَیں ݙِہدا رہ ویںداں جو ہُݨ تُہاݙے والِد دا رَوِیّہ میݙے نال پہلے واںگُوں نَئِیں رہ ڳیا، پر جیکُوں مَیں اتے میݙا والِد منیںدے ہَیں او خُدا میݙے نال ہے پیا۔ تُسّاں چںڳی طرح ڄاݨدیاں ہِوّے جو جِتنا کر سڳدا ہم مَیں تُہاݙے والِد دی خِدمت کِیتی ہے۔ پر تُہاݙا والِد میݙے نال چںڳاں نَئِیں وَرتیا۔ اُوں ݙاہ دفعہ میݙا مُعاوضہ تبدِیل کِیتے، پر رَبُّ العِزَّت اُوں کنُوں میݙا نُقصان نَئِیں تِھیوَݨ ݙِتّا۔ جے اُوں آکِھئے: ’بَربَرے تیݙا مُعاوضہ ہِن‘ تاں سارا مال بَربَرے ڄَݨَݨ پئے ڳَے اتے جے اُوں آکِھئے: ’گَدرے تیݙا مُعاوضہ ہِن‘ تاں سارا مال گَدرے ڄَݨَݨ پئے ڳَے۔ اِیں طرح رَبُّ العِزَّت اپݨی قُدرت نال تُہاݙے والِد دا مال میکُوں ݙے ݙِتّے۔ اجَڑ دے لڳَّݨ والے ݙِیہاڑِئیں مَیں خواب ݙِٹّھا جو اجَڑ کُوں بَربَرے اتے گَدرے چھیلے لَیںدے پئے ہِن۔ تݙاہُوں خواب وِچ ہِک فرِشتے میکُوں سݙ ماریا: ’یعقُوبؑ!‘ مَیں جواب ݙِتّا: ’جی سئِیں!‘ اُوں آکھیا: ’رَبُّ العِزَّت فرمیںدے: ”غَور نال ݙیکھ! جو اجَڑ کُوں لاوَݨ والے ایہ سارے چھیلے بَربَرے اتے گَدرے ہِن کِیُوں جو جیرھا کُجھ لابن تیݙے نال کریںدا رہ ڳَے مَیں او سب کُجھ ݙِہدا رہ ڳِیاں۔ مَیں اُوہو خُدا ہاں جَیں بَیت ایل وِچ تیکُوں ݙِکھالی ݙِتّی ہئی جِتّھاں تَیں پتَّھر کُوں کھڑا کر کے اُوںدے سِرے تے زَیتُون دا تیل سٹیا اتے جِتّھاں تَیں میݙے نال قَول اِقرار کِیتا۔ بَس ہُݨ جلدی نال اِتُّھوں نِکلَݨ دی کر اتے اپݨے وطن دو ول وںڄ۔“‘“ تاں بی بی راحِیلہ اتے بی بی لیاہ اُنہیں کُوں آکھیا: ”کیا ساݙے والِد دے گھر وِچُّوں ساݙا کُوئی حقّ کَینی؟ کیا اُوں ساکُوں بیگانِئیں واںگُوں نَئِیں سمجِھیا؟ جو ساکُوں ویچ کے ساݙا مُل وی کھا ڳَے۔ اِیں واسطے جیرھی دَولَت رَبُّ العِزَّت ساݙے والِد کنُوں کَھس کے ساکُوں ݙِتّی ہے او ساݙی اتے ساݙے ٻالیں دی ہے۔ بَس جِیویں رَبُّ العِزَّت تیکُوں فرمائے اُوویں کر۔“
\p تاں حضرت یعقُوبؑ تیاری کر کے اپݨے ٻالیں ٻَچّیں کُوں اُٹّھیں تے چڑھایا۔ اپݨے سارے سامان اتے مال سمیت جیرھا اُنہیں فَدّان ارام وِچ کَٹّھا کر گِھدا اپݨے والِد حضرت اِسحاقؑ دے کولھ وںڄَݨ واسطے مُلکِ کنعان دو روانہ تھی پئے۔
\p روانہ تِھیںدے ویلے بی بی راحِیلہ اپݨے والِد دے گھر وِچُّوں خاندانی مجسمے چوری چاتی ڳئی کِیُوں جو لابن اُوں ویلے اپݨے بھیݙیں دی اُن کَترواوَݨ واسطے ٻاہر ڳیا ہویا۔ حضرت یعقُوبؑ اپݨے سَوہرے لابن کُوں ݙَسِّیے بغَیر چوری چوری نِکِل ڳَے۔ اُنہیں جلدی جلدی اپݨے سارے مال مَݙّی سمیت دریائے فرات پار کر کے جِلعاد دے پہاڑی عِلاقے والے پاسے رُخ کِیتا۔

\s لابن دا حضرت یعقُوبؑ دے پِچّھو آوَݨ
\p لابن کُوں ترِیجھے ݙِیںہہ وںڄ پتّا لڳا جو حضرت یعقُوبؑ چوری نِکِل ڳَے ہِن۔ تاں اُوں اپݨے رِشتےداریں کُوں نال گِھن کے حضرت یعقُوبؑ دا پِیچھا کِیتا اتے ستویں ݙِیںہہ جِلعاد دے پہاڑی عِلاقے وِچ اُنہیں دے نزدِیک وںڄ پہُںچا۔ پر رَبُّ العِزَّت رات کُوں خواب وِچ لابن کُوں فرمایا: ”خبردار! تُوں یعقُوبؑ نال کُوئی بَد‌سلُوکی نہ کرِیں۔“ لابن حضرت یعقُوبؑ دے کولھ وںڄ پہُںچا۔ حضرت یعقُوبؑ جِلعاد دے پہاڑی عِلاقے وِچ اپݨا دیرہ لئی پئے ہَن۔ اِیں واسطے لابن وی اپݨے رِشتےداریں سمیت اُتھائِیں وںڄ دیرہ لایا۔ لابن حضرت یعقُوبؑ کُوں آکھیا: ”تَیں میݙے نال ایہ کیا کِیتے؟ میکُوں دھوکا ݙے کے میݙِئیں دِھیرِیں کُوں اِیویں گِھدّی اَئِیں جِیویں جںگ وِچُّوں تیݙے ہتّھیں آیاں ہووِن! تیکُوں چوری نِکلَݨ دی کیا ضرُورت ہئی؟ تُوں میکُوں ݙَسّیں ہا تاں مَیں تیݙے سِہرے ڳوا کے، دول دمامیں وَڄوا کے تیکُوں خُوشی نال روانہ کراں ہا۔ کیا تَیں ایہ عقل والا کم کِیتے جو ویںدے ویلے میکُوں اپݨِئیں دِھیرِیں اتے ݙوہترِئیں دے سِر وی نِوِّھی چُمَّݨ ݙِتّے؟ اِیں ویلے مَیں تیݙے نال جو کُجھ چاہاں کر سڳداں، پر کَل رات اُوں خُدا جیکُوں تُوں اتے تیݙا والِد منیںدے ہِوّے میکُوں آکِھئے: ’خبردار! تُوں یعقُوبؑ دے نال کُوئی بَد‌سلُوکی نہ کرِیں۔‘ خَیر، ہُݨ تاں تُوں ویںدا پَئِیں کِیُوں جو اپݨے والِد دے گھر واسطے ٻہُوں مُوںجھا ہَئِیں، پر تُوں ایہ ݙسا جو میݙے خاندانی مجسمے کِیُوں میݙے گھر وِچُّوں چوری چاتی اَئِیں؟“ تاں حضرت یعقُوبؑ ٻولِئے: ”چوری تاں مَیں اِیں واسطے نِکلِیاں کِیُوں جو میݙے دِل وِچ ݙر ہا۔ مَیں سوچِئے متّاں تُسّاں زبردستی اپݨِیاں دِھیرِیں مَیں کنُوں کَھس گِھنو۔ باقی جے تُہاݙے خاندانی مجسمے اِتّھاں کہِیں دے کولھ وی لبھ پووِن تاں اُوںدا خُون تُہاݙے اُتّے روا ہے۔ اِنہیں سارے بِھراویں دے سامݨے ݙیکھ گِھنو۔ جے تُہاݙی کُوئی وی چِیز میݙے کولُھوں نِکلے تاں چا گِھنو۔“ حضرت یعقُوبؑ کُوں پتّا کَینا جو بی بی راحِیلہ ایہ خاندانی مجسمے چوری چاتی آئی ہے۔
\p تاں لابن حضرت یعقُوبؑ، بی بی لیاہ اتے ݙُہائیں ٻانِھئیں دے خَیمِئیں وِچ وںڄ کے او خاندانی مجسمے ڳولئے، پر اُتّھاں نہ لَبھئے۔ ول او بی بی راحِیلہ دے خَیمے وِچ وَڑیا۔ پر اِیں کنُوں پہلے بی بی راحِیلہ اُنہیں خاندانی مجسمِئیں کُوں چا کے کچاوے وِچ وںڄ رکھیا اتے آپ اُنہیں دے اُتّے چڑھ ٻیٹھی۔ لابن اُنہیں دے خَیمے وِچ ڳُلیںدا رہ ڳیا۔ اُتّھو کیا لَبھݨا ہا؟ تاں بی بی راحِیلہ آکھݨ لڳی: ”ابّا سئِیں! ناراض نہ تِھواہے۔ مَیں اِنہیں ݙِیہاڑِئیں اُٹّھی کے تُہاݙو آوَݨ دے قابِل کَینی۔“ بَس لابن ڳول ڳول تھکّا، پر خاندانی مجسمے اُوکُوں کِتھاؤُں نہ لبِھئے۔
\p ول تاں حضرت یعقُوبؑ کُوں ݙاڈھے زور دا غُصّہ آ ڳیا اتے لابن دے نال زور دا جھیڑا لایونے۔ اُنہیں لابن کُوں آکھیا: ”آخِر میݙا قصُور کیا ہے؟ مَیں تیݙا کیرھا گُناہ کِیتے جو درُک تے میکُوں پِچّھو آن ہتّھ گَھتّی؟ تَیں میݙے سارے سامان دی تلاشی گِھن گِھدّی ہے۔ جیرھی چِیز چوری دی لدّھی میݙے اتے اپݨے اِنہیں سارے بِھراویں دے سامݨے آن رکھ! اِنصاف اِیہے کرِن۔ ݙیکھ! مَیں پُورے وِیہہ سال تیݙی خِدمت وِچ رہ ڳِیاں۔ کݙاہِیں تیݙی کُوئی بھیݙ یا ٻکری ترُٻّی ہئی؟ یا تیݙے اجَڑ دا کُوئی گھٹّا مَیں کھا ڳیا ہم؟ جے کُوئی بھیݙ یا ٻکری جںگلی جانوَریں چِیرپاڑ سٹّی تاں مَیں اُوکُوں تیݙے کولھ نہ چا آںدا، بلکہ اُوںدا نُقصان آپ پُورا کریںدا رہ ڳِیاں۔ جے کُوئی بھیݙ یا ٻکری ݙِیںہہ کُوں یا رات کُوں وی چوری تھی ڳئی تاں او تَیں مَیں کنُوں دَھر گِھدّی۔ میݙا حال تاں ایہ ہا جو ݙِیںہہ کُوں گرمی اتے رات کُوں سردی وِچ مَردا ہم۔ میݙِئیں اکِّھیں دی نِںدر اُݙ ڳئی ہئی۔ اِیں طرح وِیہہ سال مَیں تیݙی خِدمت وِچ گُزارِئے ہِن۔ چَوݙاں سال تیݙا اجَڑ چَرائے تیݙِئیں ݙُہائیں دِھیرِیں دی خاطِر اتے چھی سال مُعاوضے وِچ، جیکُوں ݙاہ دفعہ تُوں تبدِیل کریںدا رہ ڳَئِیں۔ جے خُدا جیکُوں میݙا ݙاݙا ابراہیمؑ منیںدا ہا، مَیں اتے میݙا والِد وی منیںدے ہَیں، میݙے نال نہ ہوںدا تاں ہُݨ تُوں میکُوں خالی ہتّھ کڈھ چھوڑیں ہا۔ رَبُّ العِزَّت میݙے ݙُکھ اتے میݙی مِحنت کُوں ݙیکھ کے کَل رات تیکُوں دڑکا ݙِتّے۔“
\c 17
\s حضرت یعقُوبؑ اتے لابن دے درمیان مُعاہِدہ
\p حضرت یعقُوبؑ کُوں لابن آکھیا: ”ایہ دِھیرِیں وی میݙِیاں ہِن اتے اِنہیں دے ٻال وی میݙے ہِن۔ ایہ اجَڑ وی میݙے ہِن، بلکہ ایہ جیرھا کُجھ وی تیکُوں نظردا پئے سب کُجھ میݙا ہے۔ پر مَیں اپݨِئیں دِھیرِیں اتے اُنہیں دے ٻالیں نال کیا کر سڳداں؟ لِہٰذا اسّاں ہُݨ آپس وِچ مُعاہِدہ کر گِھنُوں اتے اِیں مُعاہِدے دا نِشان تیݙے میݙے درمیان گواہی دا ثبُوت رہ ویسی۔“ حضرت یعقُوبؑ یاد رکَّھݨ واسطے ہِک پتَّھر کُوں چا کے کھڑا کر کھڑایا۔ ول رِشتےداریں کُوں آکھیونے: ”پتَّھر کَٹّھے کرو!“ تاں اُنہیں پتَّھر کَٹّھے کر کے ہِک بُرجی بݨائی۔ ول اُتھائِیں بُرجی دے نال ٻہہ کے روٹی کھادھونے۔ لابن اُوں بُرجی دا ناں اپݨی زبان وِچ ”یجرشاہدُوتھا“ اتے حضرت یعقُوبؑ اپݨی زبان وِچ اُوںدا ناں ”جِلعاد“ رکھیا۔ اِنہیں ݙُہائیں دا مطلب ہے: ”گواہی دی بُرجی۔“ لابن آکھیا: ”ایہ بُرجی اڄ کنُوں اڳُوں تے تیݙے میݙے درمیان گواہی دا نِشان ہوسی۔“ ہِیں واسطے اُوں بُرجی دا ناں “جِلعاد” پک ڳیا۔ اِیں جاہ دا ناں مِصفہ یعنی ”نِگہبانی دی بُرجی“ وی رکھیا ڳیا کِیُوں جو لابن آکھیا: ”ساݙے جُدا تِھیوَݨ دے بعد ﷲ سئِیں تیݙے اتے میݙے اُتّے اپݨے کرم دی نِگاہ فرمائی رکّھے۔ جے تَیں میݙِئیں دِھیرِیں کُوں ݙُکھ ݙِتّا یا اِنہیں تے پہاجِیں سَٹیونی تاں بھاویں ݙیکَھݨ والا کُوئی وی نہ ہووے، ول وی یاد رکِّھیں جو میݙی طرفُوں خُدا ݙِہدا پیا ہوسیا۔ ہُݨ اِݙّے اِیں بُرجی کُوں ݙیکھ کے یاد رکَّھݨ واسطے اُوں پتَّھر کُوں ݙیکھ جیکُوں مَیں اپݨے اتے تیݙے درمیان حدّ مُقرَّر کِیتے۔ ایہ بُرجی اتے ایہ کھڑا پتَّھر اِیں ڳالھ دی نِشانی ہوسِن جو نہ مَیں تیکُوں نُقصان پہُںچاوَݨ دے اِرادے نال اِیں ڈھیر کنُوں اڳُوں تے تیݙے پاسے وَدھساں اتے نہ تُوں میکُوں نُقصان پہُںچاوَݨ دے اِرادے نال اِیں بُرجی اتے اِیں کھڑے پتَّھر کنُوں اڳُوں تے میݙے پاسے وَدھسیں۔ تیݙے ݙاݙے ابراہیمؑ دا اتے میݙے ݙاݙے نَحُور دا خُدا ساݙے درمیان مُنصِف ہووے۔“ تاں حضرت یعقُوبؑ خُدا دی قَسم چاتی جیکُوں اُنہیں دے والِد حضرت اِسحاقؑ وی منیںدے ہَن۔ ول حضرت یعقُوبؑ اُوں پہاڑی مقام تے قُربانی دے طَور تے ہِک جانوَر ذبح کِیتا اتے رِشتےداریں کُوں روٹی کھاوَݨ دی دعوت ݙِتونے۔ روٹی کھا کے رات اُنہیں اُتھائِیں گُزاری۔ سویرےسویرے لابن اُٹّـِھیا، اپݨِئیں دِھیرِیں اتے ݙوہترِئیں دے سِر چُمِّئے اتے اُنہیں کُوں دُعائِیں ݙے کے پِچّھو تے اپݨے گھر دو روانہ تھی پیا۔
\c 18
\s وݙّے بِھرا کُوں مِلَݨ واسطے حضرت یعقُوبؑ دا تیاری کرَݨ
\p حضرت یعقُوبؑ وی اڳُوں تے روانہ تھی پئے۔ رستے وِچ فرِشتِئیں دی ہِک ٹولی نال اُنہیں دا آمݨا‌سامݨا تِھیا۔ فرِشتِئیں کُوں ݙیکھ کے آکھیونے: ”ایہ تاں رَبُّ العِزَّت دی ٹولی ہے۔“ اِیں واسطے اُوں جاہ دا ناں محنائم رکھیونے۔ ول حضرت یعقُوبؑ اِیں کِیتا جو اپݨے اڳُّوں اپݨے بِھرا عِیصَو دو مُلکِ ادوم وِچ قاصد روانہ کر ݙِتونے۔ اُنہیں کُوں سمجھا چھوڑِیونے: ”میݙے بِھرا عِیصَو کُوں اِیں اکھاہے جو تُہاݙے ٻانھے یعقُوبؑ آکِھئے: ’آقا! مَیں ہُݨ تَئِیں مامے لابن دے کولھ رہ ڳِیاں۔ میݙے کولھ ڳَوِّیں، ݙاںد، گَݙہا، بھیݙاں، ٻَکرِیاں، ٻانِھیاں اتے ٻانھے ہِن۔ مَیں ایہ پَیغام اپݨے آقا دو اِیں واسطے بِھڄیںدا پیاں جو میݙے اُتّے تُہاݙے کرم دی نِگاہ تِھیوے۔‘“
\p جیرھے ویلے قاصد اُتّھو تھی کے حضرت یعقُوبؑ دے کولھ ول آئے تاں ݙسایونے جو ”اسّاں تُہاݙے بِھرا عِیصَو دے کولُھوں تھی آئے ہَیں۔ او چار سَو آدمِئیں کُوں نال گِھن کے تُہاکُوں مِلَݨ واسطے اںدا پئے۔“ تاں حضرت یعقُوبؑ کُوں ݙاڈھا ݙر لڳَّݨ پئے ڳیا اتے ٻہُوں پریشان تھی ڳَے۔ ول اُنہیں اِیں کِیتا جو اپݨے آدمِئیں، بھیݙیں ٻَکرِئیں، ڳَوِّیں ݙاںد اتے اُٹّھیں دِیاں ݙُو ٹولیاں بݨایاں۔ اُنہیں سوچیا جو ”جے عِیصَو حملہ کر وی ݙِتّا تاں ہِک ٹولی ماری ویسی، ݙُوجھی تاں بچ کے بھڄ ویسی۔“

\s حضرت یعقُوبؑ دے ناں دی تبدِیلی
\p ہُوں رات حضرت یعقُوبؑ ݙُوہیں ذالِیں، ݙُوہیں حَرماں اتے اپݨے سارے ٻال دریائے یبُوق دے پَتَّݨ کنُوں پار پہُںچاوَݨ واسطے کَٹّھے کِیتے۔ ول اُنہیں سارِئیں کُوں نال گِھن کے دریا پار کرا آئے اتے جو کُجھ اُنہیں دے کولھ ہا او سب کُجھ وی پار ٻِھڄوا ݙِتونے۔ حضرت یعقُوبؑ اُتّھاں کلّھے رہ ڳَے اتے پرّھا پُھٹَّݨ تَئِیں کُوئی شخص آ کے اُنہیں دے نال کُشتی لڑدا رہ ڳیا۔ جݙّاں اُوں شخص ݙِٹّھا جو او حضرت یعقُوبؑ کنُوں ݙاڈھا نَئِیں پیا تھی سڳدا تاں اُوں اُنہیں دی ستّھل دے مُنڈھ والی رڳ کُوں پکِّڑ کے مروڑی ݙِتّی تاں رڳ ول پا ڳئی۔ ول اُوں شخص آکھیا: ”ہُݨ میکُوں چھوڑ تاں جو مَیں وںڄاں کِیُوں جو پرّھا پُھٹدی پئی ہے۔“ پر حضرت یعقُوبؑ آکھیا: ”مَیں تیکُوں او تَئِیں کَینا وںڄَݨ ݙیساں جے تَئِیں تُوں میکُوں خَیر دی دُعا نہ ݙیسیں۔“ تاں اُوں شخص پُچھیا: ”تیݙا ناں کیا ہے؟“ اُنہیں جواب ݙِتّا: ”یعقُوبؑ۔“ تاں اُوں شخص آکھیا: ”اڳُوں تے تیݙا ناں ’یعقُوبؑ‘ نہ ہوسی، بلکہ ’اِسرائِیل‘ سݙیا ویسی یعنی ’خُدا دے نال زور‌آزمائی کرَݨ والا‘ کِیُوں جو تَیں خُدا اتے آدمِئیں نال زور‌آزمائی کِیتی اتے ݙاڈھا تھی ڳیوں۔“ تاں حضرت یعقُوبؑ آکھیا: ”مہربانی کر کے اپݨا ناں تاں ݙسا!“ اُوں آکھیا: ”تُوں مَیں کنُوں میݙا ناں کِیُوں پُچھدِیں؟“ ول اُوں حضرت یعقُوبؑ کُوں خَیر دی دُعا ݙِتّی۔
\p اِیںدے بعد حضرت یعقُوبؑ اپݨے دِل وِچ آکھیا: ”میکُوں دِیدارِ اِلٰہی نصِیب تَھے اتے زِںدہ وی بچ ڳیا ہاں۔“ اِیں واسطے اُوں جاہ دا ناں فنی‌ایل یعنی ”دِیدارِ اِلٰہی“ رکھیونے۔ ول اڄّاں حضرت یعقُوبؑ فنی‌ایل وِچُّوں گُزردے پئے ہَن تاں سِجھ اُبِھر آیا۔ او لںگڑے تھی کے ٹُردے ویںدے ہَن کِیُوں جو اُنہیں دی ستّھل دے مُنڈھ والی رڳ ول پا ڳئی ہئی۔ ہِیں وجہ کنُوں بنی اِسرائِیل اڄ تَئِیں جانوَریں دی ستّھل دے مُنڈھ والی او رڳ نَئِیں کھںدے کِیُوں جو حضرت یعقُوبؑ دی اِیں رڳ کُوں اُوں شخص دا ہتّھ لڳا ہا۔

\s بِھراویں دا مِلَݨ
\p ویںدِئیں ویںدِئیں حضرت یعقُوبؑ جو غَور نال ݙِٹّھا تاں کیا ݙِہدِن جو عِیصَو چار سَو آدمی نال گِھن کے لڳا اںدے۔ تاں اُنہیں بی بی لیاہ اتے بی بی راحِیلہ اتے ݙُہائیں حَرمیں دے ٻالیں کُوں اُنہیں دِئیں مَیرِیں دے نال مِلایا۔ ول ٻانِھئیں اتے اُنہیں دے ٻالیں کُوں سارِئیں کنُوں اڳُّوں ٹوریونے۔ اُنہیں دے پِچّھو بی بی لیاہ اتے اُنہیں دے ٻالیں کُوں اتے سارِئیں کنُوں پِچّھو بی بی راحِیلہ اتے حضرت یُوسفؑ کُوں ٹوریونے۔ پر آپ اُنہیں سارِئیں دے اڳُّوں اڳُّوں ٹُردے ڳَے۔ جیرھے ویلے اپݨے بِھرا دے نزدِیک پہُںچے تاں اُوں تَئِیں پہُںچدِئیں پہُںچدِئیں ست دفعہ جُھک کے اُوکُوں سلام کِیتونے۔ عِیصَو درُک کے حضرت یعقُوبؑ کُوں آن مِلیا اتے گُھٹ کے ڳلکڑی پاتُس۔ اُنہیں کُوں چُمدا وی رہ ڳیا اتے روںدا وی رہ ڳیا۔ ول عِیصَو جیرھے ویلے سِر چا کے عَورتیں اتے ٻالیں کُوں ݙِٹّھا تاں پُچِّھیُس: ”ایہ تیݙے نال کَون ہِن؟“ حضرت یعقُوبؑ جواب ݙِتّا: ”ایہ ٻال ٻَچّے او ہِن جیرھے رَبُّ العِزَّت اپݨے فضل نال تیݙے ٻانھے کُوں بَخشِن۔“ تاں ٻانِھیاں اتے اُنہیں دے ٻال نزدِیک آئے اتے جُھک کے عِیصَو کُوں سلام کِیتونے۔ ول بی بی لیاہ اپݨے ٻالیں سمیت نزدِیک آئی اتے اُنہیں وی جُھک کے سلام کِیتا۔ آخِر وِچ بی بی راحِیلہ اتے حضرت یُوسفؑ نزدِیک آئے اتے جُھک کے سلام کِیتونے۔ ول عِیصَو حضرت یعقُوبؑ کُوں آکھیا: ”اِتنا ڈھیر سارا مال میݙو بھیڄَݨ دی کیا ضرُورت ہئی؟“ اُنہیں جواب ݙِتّا: ”تاں جو میݙے اُتّے تُہاݙے کرم دی نِگاہ تِھیوے۔“ تاں عِیصَو آکھیا: ”میݙا بِھرا! میݙے کولھ سب کُجھ ٻہُوں ہے۔ تیݙا مال تیکُوں بخشِیا ہویا ہووے۔“ ول حضرت یعقُوبؑ آکھیا: ”نہ سئِیں! اِیویں نہ کرو۔ جے میݙے اُتّے کرم دی نِگاہ تھی سڳے تاں میݙا نذرانہ قبُول کر گِھنو، کِیُوں جو تُہاکُوں اِتنا مہربان ݙیکھ کے میکُوں اِیویں پیا لڳدے جو رَبُّ العِزَّت میݙے اُتّے مہربان تھی ڳَے۔ مہربانی فرما کے میݙا نذرانہ جیرھا تُہاݙی خِدمت وِچ پیش کِیتا ڳَے قبُول کر گِھنو کِیُوں جو رَبُّ العِزَّت دا میݙے اُتّے وݙّا فضل ہے اتے اُوںدا ݙِتّا میݙے کولھ ٻہُوں کُجھ ہے۔“ اِیویں حضرت یعقُوبؑ جیرھے ویلے ڈھیر ساری مِنَّت کِیتی تاں عِیصَو او نذرانہ قبُول کر گِھدا۔ ول عِیصَو آکھیا: ”آ! ہُݨ پَںدھ کرُوں۔ مَیں تیکُوں اپݨے نال گِھن ڄُلداں۔“ پر حضرت یعقُوبؑ آکھیا: ”سئِیں! تُہاکُوں تاں پتّا ہے جو میݙے نال چھوٹے چھوٹے ٻال ہِن اتے کِھیر پِلاوَݨ والِئیں بھیݙیں، ٻَکرِئیں اتے ڳَوِّیں دی سںبھال وی رکھݨی ہے۔ جے اِنہیں کُوں سارا ݙِیںہہ ہَکلی ڄُلُوں تاں سارے ݙَںڳَر ڈھور مَر پوسِن۔ سئِیں! میݙی عرض تاں اِیویں ہے جو تُسّاں اپݨے ٻانھے کنُوں اڳُّوں ڄُلّو پئے۔ مَیں آہستہ آہستہ آؤساں پیا جِیویں جِیویں ایہ ݙَںڳَر ڈھور اتے ٻال بَرداشت کر سڳسِن۔ آخِر مَیں تُہاݙے کولھ ادوم وِچ پہُںچ ویساں۔“ تاں عِیصَو آکھیا: ”جے ضرُورت ہووے تاں مَیں اپݨے آدمِئیں وِچُّوں کُجھ تیݙے کولھ کھڑائی وںڄاں؟“ حضرت یعقُوبؑ آکھیا: ”بَس سئِیں! تُہاݙی مہربانی! میݙے واسطے اِیہو کُجھ ٻہُوں ہے جو مَیں تُہاݙا منظُورِ نظر تھی پیاں۔“ تاں عِیصَو ہُوں ݙِیںہہ پِچّھو تے ادوم دو ول ڳیا۔ پر حضرت یعقُوبؑ ٻے پاسے رُخ چا کِیتا اتے سفر کریںدِئیں ہوئیں سُکات وِچ وںڄ پہُںچے۔ اُتّھاں اُنہیں اپݨے واسطے ہِک گھر بݨایا اتے اپݨے ݙَںڳَر ڈھوریں واسطے چَھپَّر بݨائے۔ ہِیں وجہ کنُوں اُوں جاہ دا ناں سُکات یعنی ”چَھپَّر“ پئے ڳیا۔
\p حضرت یعقُوبؑ فَدّان ارام کنُوں آ کے مُلکِ کنعان دے شہر شکِیم دے نزدِیک صحِیح سلامت پہُںچ ڳَے اتے شہر دے مشرقی پاسے اپݨا دیرہ لا ݙِتونے۔ زمِین دے جیرھے ٹُکڑے تے حضرت یعقُوبؑ اپݨے خَیمے لَے ہَن او ٹُکڑا اُنہیں شکِیم دے والِد حمُور دے خاندان کنُوں چاںدی دے سَو سِکّے ݙے کے مُل گِھن گِھدا۔ اُنہیں اُتّھاں بارگاہِ اِلٰہی وِچ قُربانِیاں پیش کرَݨ واسطے ہِک تَھلّا بݨایا اتے اُوکُوں رَبُّ العِزَّت دے ناں تے وقف کر ݙِتونے۔

\s دِینہ دی عِزَّت دا لُٹِّیا وںڄَݨ
\p حضرت یعقُوبؑ دی دِھی دِینہ جیرھی بی بی لیاہ کنُوں ڄَئی ہئی شکِیم شہر دِئیں چھوہرِیں کُوں مِلَݨ واسطے ڳئی۔ اُوں عِلاقے دے حاکِم حمُور حُوری دے پُتر شکِیم اُوکُوں ݙِٹّھا تاں اُوکُوں پکِّڑ کے گھر گِھن ڳیا اتے اُوںدی عِزَّت لُٹ گِھدُّس۔ دِینہ نال اُوںدا دِل لڳ ڳیا۔ او اُوکُوں پیار کرَݨ لڳ پیا اتے اُوکُوں دِلاسا ݙِتُّس۔ اِیںدے بعد شکِیم اپݨے پِئُو کُوں آکھیا: ”ہِیں چھوہِر نال میݙی شادی کرواؤ۔“ جیرھے ویلے حضرت یعقُوبؑ کُوں پتّا لڳا جو شکِیم میݙی دِھی دِینہ دی عِزَّت لُٹ گِھدّی ہے اتے اُوکُوں قابُو کِیتی پئے تاں اُوں ویلے اُنہیں دے پُتر ݙَںڳَر ڈھوریں نال جںگل وِچ دُور ڳَے ودّے ہَن۔ ہِیں واسطے حضرت یعقُوبؑ اُنہیں دے گھر آوَݨ تَئِیں چُپ کِیتی رہ ڳَے۔ اِیں دَوران حمُور اپݨے پُتر شکِیم دے نال حضرت یعقُوبؑ دے کولھ دِینہ دے سِلسلے وِچ ڳالھ مُہاڑ کرَݨ واسطے آ ڳیا۔ تاں اُوں ویلے حضرت یعقُوبؑ دے پُتر وی جںگل کنُوں گھر ول آئے ہَن۔ دِینہ دے بارے وِچ سُݨ کے اُنہیں کُوں ݙاڈھا ݙُکھ تِھیا اتے اُنہیں دے تن بَدن وِچ بھا لڳ ڳئی کِیُوں جو شکِیم حضرت یعقُوبؑ دی دِھی دی عِزَّت لُٹ کے بنی اِسرائِیل کُوں ذلِیل و خُوار کرَݨ والا کم کِیتا جیرھا ہرگِز مُناسِب کَینا۔ اُتھائِیں حمُور اُنہیں کُوں آکھݨ لڳ پیا: ”میݙا پُتر شکِیم دِینہ نال تہ دِلُوں شادی کرَݨ چاہںدے۔ تُسّاں مہربانی کرو۔ ساݙے نال رِشتے داری کر گِھنو۔ ساݙِیاں دِھیرِیں تُسّاں گِھنو اتے اپݨِیاں دِھیرِیں ساکُوں ݙیوو۔ تُسّاں ساݙے نال اِتھائِیں رہ وںڄو۔ اِیں عِلاقے کُوں اپݨا سمجھو۔ ہِیںدے وِچ رہ کے سوداگری کرو اتے اِتھائِیں اپݨی ملکِیَّت بݨاؤ۔“ شکِیم وی حضرت یعقُوبؑ اتے اُنہیں دے پُتریں کُوں آکھیا: ”میݙے اُتّے بَس تُہاݙے کرم دی نِگاہ تھی پووے۔ تُسّاں جِیویں آکھسو، مَیں اُوویں کریساں۔ جِتنا وی حقّ مَہر تُسّاں آکھسو، مَیں ݙیساں۔ پر دِینہ دی شادی میݙے نال کر ݙیوو۔“ تاں حضرت یعقُوبؑ دے پُتریں شکِیم اتے اُوںدے پِئُو حمُور کُوں چالاکی نال جواب ݙِتّا کِیُوں جو شکِیم اُنہیں دی بھیݨ دی عِزَّت لُٹ گِھدّی ہئی۔ اُنہیں آکھیا: ”ہِک اݨ طہورے شخص کُوں اسّاں اپݨی بھیݨ کِیویں ݙے سڳدے ہَیں؟ ایہ تاں ساݙے واسطے ٻہُوں وݙّی بَد‌نامی ہے۔ ، البتَّہ اسّاں ہِک شرط تے راضی تھی سڳدے ہَیں جے تُسّاں اسّاں واںگُوں تھی وںڄو یعنی تُہاݙے وی ہر مرد دی طہور کر ݙِتّی وںڄے۔ ول اسّاں ہِک ٻے کُوں دِھیرِیں ݙیسُوں گِھنسُوں، تُہاݙے مُلک وِچ رہ ویسُوں اتے تُہاݙے نال ہِکّا قَوم بݨ ویسُوں۔ جے تُسّاں طہور کراوَݨ والی ساݙی ڳالھ نہ مَنّی تاں ول اسّاں اپݨی بھیݨ کُوں گِھن کے اِتُّھوں لڳے ویسُوں۔“
\p اُنہیں دی ایہ تجوِیز حمُور اتے اُوںدے پُتر شکِیم کُوں پسںد آئی۔ بَس اُوں جَوان اِیں کم وِچ دیر نہ کِیتی کِیُوں جو او دِینہ تے فِدا ہا۔ او اپݨے قبِیلے وِچ سارِئیں کنُوں وݙّی عِزَّت والا ہا۔ تاں حمُور اتے اُوںدا پُتر شکِیم اپݨے شہر دے پھاٹک تے کچہری وِچ ڳَے اتے اپݨے لوکیں کُوں آکھیونے: ”یعقُوبؑ دا خاندان ساݙے نال میل ڄول رکَّھݨ چاہںدے۔ بَس او اِیں عِلاقے وِچ رہ وںڄِن اتے سوداگری کرِن کِیُوں جو اِنہیں دے رہَݨ واسطے اِیں عِلاقے وِچ جاہ ٻہُوں ہے اتے اسّاں ہِک ٻے کُوں دِھیرِیں ݙیسُوں گِھنسُوں۔ ایہ لوک وی ساݙے نال رہ وںڄَݨ اتے ہِکّا قَوم بݨ وںڄَݨ تے راضی ہِن۔ صِرف اِیں شرط تے جو ساݙے ہر ہِک مرد دی طہور کر ݙِتّی وںڄے جِیویں اُنہیں دی تھئی ہُوئی ہے۔ بَس جے اسّاں اُنہیں دی اِیہا ہِک ڳالھ منّ گِھنُوں تاں او ہمیشہ ساݙے نال رہ ویسِن۔ ول بھلا اُنہیں دے اجَڑ، ݙَںڳَر ڈھور اتے مال مَݙّی، ایہ سب کُجھ ساݙا نہ تھی ویسی؟“ تاں اُوں شہر دے سارے لوکیں حمُور اتے اُوںدے پُتر شکِیم دی ایہ ڳالھ منّ گِھدّی اتے سارِئیں اپݨی طہور کرائی۔
\p طہور دے ترِیجھے ݙِیںہہ جݙّاں او ݙاڈھے درد وِچ مُبتلا ہَن تاں اِیں تِھیا جو حضرت یعقُوبؑ دے پُتریں وِچُّوں دِینہ دے ݙُو سَکّے بِھرا شِمعُون تے لیوی اپݨی اپݨی تلوار چا کے بے دھڑک شہر وِچ وَڑ ڳَے اتے اُتّھو دے سارے مَردیں کُوں قتل کر ݙِتونے۔ اُنہیں حمُور اتے اُوںدے پُتر شکِیم کُوں وی تلوار دی دھار نال مار سٹیا اتے دِینہ کُوں شکِیم دے گھر کنُوں اپݨے نال گِھن ڳَے۔ اُتّھو دے لوکیں کُوں قتل کرَݨ دے بعد ول حضرت یعقُوبؑ دے پُتریں اُوں شہر دا سارا مال لُٹ گِھدا جِتّھاں اُنہیں دی بھیݨ دی عِزَّت لُٹّی ڳئی ہئی۔ یعنی اُنہیں دِیاں بھیݙاں ٻَکرِیاں، ڳَوِّیں ݙاںد، گَݙہا، ساری دَولَت اتے ٻیا جو کُجھ وی اُنہیں دے گھریں وِچ یا ٻاہر زمِینیں تے، سب کُجھ لُٹ گِھدونے۔ اُنہیں دِئیں سارِئیں عَورتیں کُوں وی ٻالیں سمیت پکِّڑ کے اپݨے نال گِھن ڳَے۔
\p شِمعُون اتے لیوی کُوں حضرت یعقُوبؑ آکھیا: ”تُسّاں میکُوں کِتھاؤُں دا نَئِیں چھوڑیا کِیُوں جو ہُݨ اِیں مُلک دے رہَݨ والے کنعانی اتے فَرِزِّی مَیں کنُوں نفرَت کریسِن۔ اسّاں تاں تھوڑے جہیں ہَیں۔ جے اُنہیں سارِئیں مِل کے میݙے اُتّے حملہ کر ݙِتّا تاں میکُوں گھر‌ٻار سمیت تباہ و برباد کر ݙیسِن۔“ او ٻولِئے: ”کیا اُوکُوں ایہ مُناسِب ہا جو ساݙی بھیݨ نال اِیویں سلُوک کرے جِیویں پیشہ وَر عَورت نال کِیتا ویںدے؟“
\c 22
\s حضرت یعقُوبؑ دا بَیت ایل دو واپس وںڄَݨ
\p حضرت یعقُوبؑ کُوں رَبُّ العِزَّت فرمایا: ”بَیت ایل دو لڳا وںڄ اتے اُتھائِیں رہ۔ میݙی بارگاہ وِچ قُربانِیاں پیش کرَݨ واسطے اُتّھاں ہِک تَھلّا بݨاوِیں۔ مَیں اُوہو ہاں جَیں تیکُوں اُوں ویلے ݙِکھالی ݙِتّی ہئی جیرھے ویلے تُوں اپݨے بِھرا عِیصَو دے ݙر کنُوں بَھڄِّیا ویںدا ہاویں۔“ تاں حضرت یعقُوبؑ اپݨے گھراݨے اتے اپݨے سارے لوکیں کُوں آکھیا: ”جَیںدے جَیںدے کولھ وی خاندانی مجسمے ہووِن کڈھ کے بُھکا سَٹّے۔ اپݨے آپ کُوں پاک صاف کرو اتے اپݨے کپڑے وَٹاؤ۔ ول اسّاں اِتُّھوں روانہ تھی کے بَیت ایل دو ڄُلُوں۔ اُتّھاں مَیں بارگاہِ اِلٰہی وِچ قُربانِیاں پیش کرَݨ واسطے ہِک تَھلّا بݨیساں جَیں اَوکھے ویلے میݙی مدَد کِیتی اتے جیرھا ہر جاہ تے میݙے نال رہ ڳَے۔“ تاں جَیںدے جَیںدے کولھ وی خاندانی مجسمے ہَن اُنہیں سارِئیں او کڈھ کے اتے کنّیں دے مُںدرے لہا کے حضرت یعقُوبؑ کُوں ݙے ݙِتّے۔ اُنہیں ایہ سب کُجھ اُوں شاہ بَلُوط دے درخت دے تلّے پُور چھوڑیا جیرھا شکِیم شہر دے نزدِیک ہا۔
\p اِیںدے بعد حضرت یعقُوبؑ اُتّھو کُوچ کِیتا۔ آسے پاسے دے شہریں وِچ وَسَّݨ والے لوکیں تے رَبُّ العِزَّت دا اِہجا خَوف طاری تھی ڳیا جو کہِیں وی اُنہیں دا پِیچھا نہ کِیتا۔ اِیں طرح حضرت یعقُوبؑ اتے او سارے لوک جیرھے اُنہیں دے نال ہَن مُلکِ کنعان دی وستی لُوز یعنی بَیت ایل وِچ آ ڳَے۔ حضرت یعقُوبؑ بارگاہِ اِلٰہی وِچ قُربانِیاں پیش کرَݨ واسطے اُتّھاں ہِک تَھلّا بݨایا اتے اُوں جاہ کُوں ایل‌بَیت ایل یعنی ”بَیت ایل وِچ ظاہِر تِھیوَݨ والے خُدا دی جاہ“ سݙیونے کِیُوں جو جݙّاں او اپݨے بِھرا دے ݙر کنُوں بَھڄِّیے ویںدے ہَن تاں ہِیں جاہ تے رَبُّ العِزَّت اُنہیں کُوں ݙِکھالی ݙِتّی ہئی۔
\p ول بعد وِچ اُتھائِیں بی بی رِفقہ کُوں پالَݨ والی دائی دبُورہ فَوت تھی ڳئی اتے اُوکُوں بَیت ایل دے جنُوبی پاسے شاہ بَلُوط دے درخت دے تلّے دفن کِیتا ڳیا۔ ہِیں وجہ کنُوں اُوں بَلُوط دا ناں ”ݙُکّھیں والا“ پک ڳیا۔ فَدّان ارام کنُوں واپس آوَݨ دے بعد جݙّاں حضرت یعقُوبؑ بَیت ایل وِچ آئے ہوئے ہَن رَبُّ العِزَّت ولا اُنہیں کُوں ݙِکھالی ݙِتّی، برکَت بخشی اتے فرمایُس: ”اڄّاں تَئِیں تاں لوک تیکُوں ’یعقُوبؑ‘ اَہدِن، پر اڳُوں تے او تیکُوں ’اِسرائِیل‘ سݙیسِن۔ مَیں قادرِ مُطلق خُدا ہاں۔ تیکُوں اِتنا وَدھیساں پَھلیساں جو تَیں کنُوں قَوم، بلکہ ڈھیر سارِیاں قَوماں بݨ ویسِن اتے تیݙی نسل وِچُّوں کئی بادشاہ بݨسِن۔ اِیہو مُلک جیرھا مَیں ابراہیمؑ اتے اِسحاقؑ کُوں بخشِیا ہا، تیکُوں ݙیساں اتے تیݙے بعد تیݙی آل اَولاد کُوں ݙیساں۔“
\p اِنہیں ڳالِھیں دے بعد حضرت یعقُوبؑ اُوں جاہ کُوں وی یاد رکَّھݨ واسطے جِتّھاں رَبُّ العِزَّت اُنہیں کُوں ہُݨ ݙِکھالی ݙِتّی ہئی ہِک پتَّھر کھڑا کر کھڑایا۔ ول اُوںدے سِرے تے آبِ اںگُور اتے زَیتُون دا تیل سَٹیونے۔ ول اُنہیں اپݨے دِل وِچ آکھیا: ”بے شک ایہ جاہ بَیت ایل یعنی ’رَبُّ العِزَّت دا گھر‘ ہے۔“
\c 23
\s بی بی راحِیلہ دا وفات پاوَݨ
\p ول حضرت یعقُوبؑ بَیت ایل کنُوں وی کُوچ چا کِیتا۔ ویںدِئیں ویںدِئیں رستے وِچ اڄّاں اِفراتہ دی وستی کنُوں کُجھ دُور ودّے ہَن تاں بی بی راحِیلہ کُوں وِیئَم دِیاں ݙاڈھے زور دِیاں پِیڑاں لڳ پیاں۔ ٻال ڄاوَݨ وِچ ٻہُوں دِقّت تِھیںدی پئی ہئی۔ ٻال ڄَݨیںدِئیں ہوئیں جݙّاں اُنہیں کُوں ݙاڈھے زور دی آخری پِیڑ لڳی پئی ہئی تاں دائی آکھیا: ”گھٻرا نہ! ایہ وی پُتر ہِوّی۔“ ہِیں دَوران بی بی راحِیلہ دا ساہ نِکلَݨ پئے ڳیا۔ مَر مَسائیں اپݨے اُوں پُتر دا ناں بِن اُونی یعنی ”ݙُکّھیں ݙِتّا“ رکھیُس۔ اِیںدے نال بی بی وفات پا ڳئی۔ پر حضرت یعقُوبؑ اُوں پُتر دا ناں بِنیامِین یعنی ”بَختاور پُتر“ رکھ ݙِتّا۔
\p بی بی راحِیلہ کُوں اِفراتہ یعنی بَیت لَحَم دے رستے وِچ دفن کِیتا ڳیا۔ حضرت یعقُوبؑ اُنہیں دی قبر تے ہِک بُرجی بݨا کھڑائی جیرھی اڄ تَئِیں مَوجُود ہے۔ ول حضرت یعقُوبؑ اُتّھو کُوچ چا کِیتا اتے عادر دے بُرج دے جنُوبی پاسُوں اپݨے خَیمے وںڄ لایونے۔ حضرت یعقُوبؑ اُتھائِیں رہَݨ لڳ پئے۔ ہِک ݙِیںہہ اِیں تِھیا جو روبَن اپݨی مترائی ماء بی بی بِلہہ دے نال اپݨا مُوںہہ کالا کِیتا اتے حضرت یعقُوبؑ کُوں اِیں ڳالھ دا پتّا وی لڳ ڳیا۔
\p حضرت یعقُوبؑ دے ٻارَھاں پُتر ہَن۔ بی بی لیاہ کنُوں روبَن جیرھا حضرت یعقُوبؑ دا وݙّا پُتر ، ول شِمعُون، لیوی، یہُودَہ، اِشکار اتے زبُلُون پَیدا تَھے۔ بی بی راحِیلہ کنُوں حضرت یُوسفؑ اتے بِنیامِین پَیدا تَھے۔ بی بی راحِیلہ دی ٻانھی بی بی بِلہہ کنُوں دان اتے نفتالی پَیدا تَھے۔ بی بی لیاہ دی ٻانھی بی بی زِلفہ کنُوں جاد اتے آشر پَیدا تَھے۔ اِیہے ہِن حضرت یعقُوبؑ دے پُتر جیرھے فَدّان ارام وِچ پَیدا تَھے۔
\c 24
\s حضرت اِسحاقؑ دا اِنتقال فرماوَݨ
\p اِیںدے بعد حضرت یعقُوبؑ مَمَر دے مقام تے جیرھا قریت‌اَربَع یعنی الخَلِیل دے نزدِیک ہے آ کے اپݨے والِد حضرت اِسحاقؑ دے کولھ رہَݨ لڳ پئے۔ اِتھائِیں پہلے حضرت ابراہیمؑ اتے ہُݨ حضرت اِسحاقؑ رہںدے ہَن۔ اِتھائِیں حضرت اِسحاقؑ ہِک سَو اَسّی سال دی عُمر وِچ بِالکُل ٻُڈھڑے تھی کے سکُون نال اِنتقال فرمایا۔ اُنہیں دے پُتریں عِیصَو اتے حضرت یعقُوبؑ مِل کے اُنہیں کُوں دَفنایا۔
\c 25
\s عِیصَو دی آل اَولاد
\p عِیصَو جیکُوں ادوم وی آکھیا ویںدے اُوںدی آل اَولاد ایہ ہئی۔ عِیصَو جیرِہئیں ݙُو کنعانی چھوہرِیں نال شادِیاں کِیتِیاں ہَن اُنہیں وِچُّوں ہِک دا ناں عادَہ ہا۔ او ایلون حِتّی دی دِھی ہئی اتے ݙُوجھی دا ناں اُہلِیبامہ ہا۔ او عنہ دی دِھی اتے صِبعون حُوری دی پوتری ہئی۔ اِنہیں دے بعد عِیصَو، حضرت اِسماعِیلؑ دی دِھی بشامہ نال شادی کِیتی جیرھی نبایوط دی بھیݨ ہئی۔ عادَہ کنُوں عِیصَو دا پُتر اَلِیفاز اتے بشامہ کنُوں رَعُوایل ڄایا۔ اُہلِیبامہ کنُوں یعُوس، یعلام، اتے قورح پَیدا تَھے۔ اِیہے عِیصَو دے پُتر ہَن جیرھے کنعان دے مُلک وِچ پَیدا تَھے۔
\p ول عِیصَو اِیں کِیتا جو اپݨِئیں ذالِیں، پُتریں، دِھیرِیں، نَوکریں چاکریں، ݙَںڳَر ڈھوریں اتے اپݨی ساری مال مَݙّی سمیت جیرھی اُوں مُلکِ کنعان وِچ کٹّھی کِیتی ہئی اپݨے بِھرا حضرت یعقُوبؑ کنُوں دُور شعِیر دے پہاڑی عِلاقے دو لڳا ڳیا، کِیُوں جو اُنہیں دے کولھ مال مَݙّی اِتنی ڈھیر تھی ڳئی ہئی جو او ݙُوہیں ہِکّی جاہ تے نہ رہ سڳدے ہَن۔ او جِتّھاں رہںدے پئے ہَن اُتّھاں اِتنے ڈھیر سارے ݙَںڳَر ڈھوریں دا گُزارہ نہ تھی سڳدا ہا۔ ہِیں خاطِر عِیصَو یعنی ادوم شعِیر دے پہاڑی عِلاقے وِچ وںڄ وَسیا۔
\p شعِیر دے پہاڑی عِلاقے والِئیں ادومِئیں دے وݙّے عِیصَو دی آل اَولاد ایہ ہئی: عِیصَو دی پہلی ذال عادَہ کنُوں اَلِیفاز ڄایا۔ اَلِیفاز دے پُتر ایہ ہَن: تِیمان، اُومر، صفُو، جعتام، اتے قنز۔ عِیصَو دے پُتر اَلِیفاز دی ہِک حَرَم وی ہئی جَیںدا ناں تِمنَع ہا۔ اُوں کنُوں وی اَلِیفاز دا پُتر عمالِیق ڄایا۔ اِیہے ہَن عِیصَو دی ذال عادَہ دے پوترے۔
\p عِیصَو دی ذال بشامہ کنُوں رَعُوایل ڄایا۔ رَعُوایل دے پُتر ایہ ہَن: نحت، زارح، شمّہ، اتے مِزّہ۔ اِیہے ہَن عِیصَو دی ذال بشامہ دے پوترے۔
\p عِیصَو دی ذال اُہلِیبامہ جیرھی عنہ دی دِھی اتے صِبعون حُوری دی پوتری ہئی، اُوں کنُوں عِیصَو دے پُتر یعُوس، یعلام، اتے قورح پَیدا تَھے۔ بنی‌عِیصَو وِچُّوں قبِیلِئیں دے سردار ایہ بݨِئے: عِیصَو دے وݙّے پُتر اَلِیفاز دی اَولاد: سردار تِیمان، سردار اُومر، سردار صفُو، سردار جعتام، سردار قنز، اتے سردار عمالِیق۔ ادوم دے مُلک وِچ اَلِیفاز دے اِیہے پُتر قبِیلِئیں دے سردار ہَن۔ ایہ سارے عادَہ دے پوترے ہَن۔
\p ادوم دے مُلک وِچ رَعُوایل دی اَولاد وِچُّوں جیرھے سردار بݨِئے ایہ ہَن: سردار نحت، سردار زارح، سردار شمّہ، اتے سردار مِزّہ۔ ایہ عِیصَو دی ذال بشامہ دے پوترے ہَن۔ عِیصَو دی ذال اُہلِیبامہ دے پُتر ایہ ہَن: سردار یعُوس، سردار یعلام، اتے سردار قورح۔ اِیہے ہَن او سردار جیرھے عنہ دی دِھی عِیصَو دی ذال اُہلِیبامہ کنُوں پَیدا تَھے۔ عِیصَو دی اَولاد یعنی بنی‌ادوم دے سردار اِیہے ہَن۔
\c 26
\s شعِیر حُوری دی آل اَولاد
\p شعِیر حُوری دی آل اَولاد جیرھی اِیں عِلاقے وِچ رہںدی ہئی اُنہیں دے ناں ایہ ہَن: لوطان، سوبل، صِبعون، عنہ، دیشون، ایصر، اتے دیشان۔ ادوم دے مُلک وِچ شعِیر دی اَولاد یعنی حُورِئیں وِچُّوں اِیہے سردار بݨِئے۔ حُوری اتے ہیمام لوطان دے پُتر ہَن، اتے تِمنَع لوطان دی بھیݨ ہئی۔ سوبل دے پُتر ایہ ہَن: علوان، مانحت، عیبال، سفُّو، اتے اونام۔ صِبعون دے پُتر ایہ ہَن: آیہ اتے عنہ۔ ایہ عنہ اُوہو ہے جَیں اپݨے والِد صِبعون دے گَݙہا چَریںدِئیں ہوئیں بیابان وِچ گرم پاݨی دے چشمے لدّھے ہَن۔ عنہ دی اَولاد ایہ ہئی: پُتر دیشون اتے دِھی اُہلِیبامہ۔ دیشون دے پُتر ایہ ہَن: حمدان، اشبان، یِتران، اتے کران۔ ایصر دے پُتر ایہ ہَن: بِلہان، زعوان، اتے عقان۔ دیشان دے پُتر ایہ ہَن: عُوز اتے اَران۔ حُورِئیں دے سردار ایہ ہَن: سردار لوطان، سردار سوبل، سردار صِبعون، سردار عنہ، سردار دیشون، سردار ایصر، اتے سردار دیشان۔ اِیہے اپݨے اپݨے قبِیلے وِچ حُورِئیں دے سردار ہَن جیرھے مُلکِ شعِیر وِچ رہںدے ہَن۔
\c 27
\s مُلکِ ادوم دے بادشاہ
\p بنی اِسرائِیل دے وِچُّوں کہِیں دے وی بادشاہ بݨَݨ کنُوں پہلے مُلکِ ادوم تے جِنہیں بادشاہی کِیتی، او ایہ ہَن: بعُور دا پُتر بالَع۔ اُوںدا دارُالخِلافہ دِنہابہ ہا۔ بالَع دے مرَݨ دے بعد اُوںدی جاہ تے بَصرہ والے زارح دے پُتر یُوباب بادشاہی کِیتی۔ یُوباب دے مرَݨ دے بعد اُوںدی جاہ تے تِیمانِئیں دے مُلک والے حُشام بادشاہی کِیتی۔ حُشام دے مرَݨ دے بعد اُوںدی جاہ تے بِدد دے پُتر ہدد بادشاہی کِیتی جَیں مُلکِ موآب وِچ مِدیانِئیں دے اُتّے فتح پاتی ہئی۔ اُوںدا دارُالخِلافہ عَوِیت ہا۔ ہدد دے مرَݨ دے بعد اُوںدی جاہ تے مسریقہ والے شَملے بادشاہی کِیتی۔ شَملے دے مرَݨ دے بعد اُوںدی جاہ تے شاؤُل بادشاہی کِیتی جیرھا دریائے فرات دے کِنارے والے شہر رحوبوت دا رہَݨ والا ہا۔ شاؤُل دے مرَݨ دے بعد اُوںدی جاہ تے عکبور دے پُتر بعل‌حنّان بادشاہی کِیتی۔ عکبور دے پُتر بعل‌حنّان دے مرَݨ دے بعد اُوںدی جاہ تے ہدد بادشاہی کِیتی۔ اُوںدا دارُالخِلافہ فاعو ہا، اتے اُوںدی ذال دا ناں مہیطب‌ایل ہا جیرھی مطرید دی دِھی اتے میضاہاب دی پوتری ہئی۔
\p بنی‌عِیصَو دے سرداریں دے ناں اُنہیں دے خاندانیں اتے رہَݨ والِئیں جاہِیں دے مُطابِق ایہ ہَن: سردار تِمنَع، سردار عَلوَہ، سردار یتیت، سردار اُہلِیباما، سردار ایلا، سردار فِینون، سردار قنز، سردار تِیمان، سردار مِبصار، سردار مجدایل، اتے سردار عِرام۔ اِیہے اپݨِئیں رہَݨ والِئیں جاہِیں دے مُطابِق جِتّھاں اِنہیں قبضہ کِیتا ہا، بنی‌ادوم دے سردار بݨِئے۔ ادومِئیں دا وݙّا عِیصَو ہا۔
